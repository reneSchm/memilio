144 template <class InfectionState>
145 double result_distance_2norm(const std::vector<mio::TimeSeries<double>>& result1,
146  const std::vector<mio::TimeSeries<double>>& result2, InfectionState compartment)
147 {
148  assert(result1.size() == result2.size());
149  assert(result1.size() > 0);
150  assert(result1[0].get_num_time_points() > 0);
151  assert(result1[0].get_num_elements() > 0);
152 
153  auto num_compartments = Eigen::Index(InfectionState::Count);
154  auto num_age_groups = result1[0].get_num_elements() / num_compartments;
155 
156  auto norm_sqr = 0.0;
157  for (auto iter_node1 = result1.begin(), iter_node2 = result2.begin(); iter_node1 < result1.end();
158  ++iter_node1, ++iter_node2) {
159  for (Eigen::Index time_idx = 0; time_idx < iter_node1->get_num_time_points(); ++time_idx) {
160  auto v1 = (*iter_node1)[time_idx];
161  auto v2 = (*iter_node2)[time_idx];
162  for (Eigen::Index age_idx = 0; age_idx < num_age_groups; ++age_idx) {
163  auto d1 = v1[age_idx * num_compartments + Eigen::Index(compartment)];
164  auto d2 = v2[age_idx * num_compartments + Eigen::Index(compartment)];
165  norm_sqr += (d1 - d2) * (d1 - d2);
166  }
167  }
168  }
169  return std::sqrt(norm_sqr);
170 }
