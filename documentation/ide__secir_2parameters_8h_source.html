174  bool check_constraints() const
175  {
176  // For parameters potentially depending on the infectious age, values are checked
177  // equidistantly on a realistic maximum window.
178  // Please note that this is an incomplete check on correctness.
179  size_t infectious_window_check = 50; // parameter defining minimal window on x-axis
180  for (size_t i = 0; i < infectious_window_check; i++) {
181  if (this->get<TransmissionProbabilityOnContact>().eval((ScalarType)i) < 0.0 ||
182  this->get<TransmissionProbabilityOnContact>().eval((ScalarType)i) > 1.0) {
183  log_error("Constraint check: TransmissionProbabilityOnContact smaller {:d} or larger {:d} at some "
184  "time {:d}",
185  0, 1, i);
186  return true;
187  }
188  }
189 
190  for (size_t i = 0; i < infectious_window_check; i++) {
191  if (this->get<RelativeTransmissionNoSymptoms>().eval((ScalarType)i) < 0.0 ||
192  this->get<RelativeTransmissionNoSymptoms>().eval((ScalarType)i) > 1.0) {
193  log_error("Constraint check: RelativeTransmissionNoSymptoms smaller {:d} or larger {:d} at some "
194  "time {:d}",
195  0, 1, i);
196  return true;
197  }
198  }
199 
200  for (size_t i = 0; i < infectious_window_check; i++) {
201  if (this->get<RiskOfInfectionFromSymptomatic>().eval((ScalarType)i) < 0.0 ||
202  this->get<RiskOfInfectionFromSymptomatic>().eval((ScalarType)i) > 1.0) {
203  log_error("Constraint check: RiskOfInfectionFromSymptomatic smaller {:d} or larger {:d} at some "
204  "time {:d}",
205  0, 1, i);
206  return true;
207  }
208  }
209 
210  for (size_t i = 0; i < (int)InfectionTransition::Count; i++) {
211  if (this->get<TransitionProbabilities>()[i] < 0.0 || this->get<TransitionProbabilities>()[i] > 1.0) {
212  log_error("Constraint check: One parameter in TransitionProbabilities smaller {:d} or larger {:d}", 0,
213  1);
214  return true;
215  }
216  }
217 
218  if (!floating_point_equal(this->get<TransitionProbabilities>()[(int)InfectionTransition::SusceptibleToExposed],
219  1.0, 1e-14)) {
220  log_error("Constraint check: Parameter transition probability for SusceptibleToExposed unequal to {:d}", 1);
221  return true;
222  }
223 
224  if (!floating_point_equal(
225  this->get<TransitionProbabilities>()[(int)InfectionTransition::ExposedToInfectedNoSymptoms], 1.0,
226  1e-14)) {
227  log_error(
228  "Constraint check: Parameter transition probability for ExposedToInfectedNoSymptoms unequal to {:d}",
229  1);
230  return true;
231  }
232 
233  if (!floating_point_equal(
234  this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedNoSymptomsToInfectedSymptoms] +
235  this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedNoSymptomsToRecovered],
236  1.0, 1e-14)) {
237  log_error("Constraint check: Sum of transition probability for InfectedNoSymptomsToInfectedSymptoms and "
238  "InfectedNoSymptomsToRecovered not equal to {:d}",
239  1);
240  return true;
241  }
242 
243  if (!floating_point_equal(
244  this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedSymptomsToInfectedSevere] +
245  this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedSymptomsToRecovered],
246  1.0, 1e-14)) {
247  log_error("Constraint check: Sum of transition probability for InfectedSymptomsToInfectedSevere and "
248  "InfectedSymptomsToRecovered not equal to {:d}",
249  1);
250  return true;
251  }
252 
253  if (!floating_point_equal(
254  this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedSevereToInfectedCritical] +
255  this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedSevereToRecovered],
256  1.0, 1e-14)) {
257  log_error("Constraint check: Sum of transition probability for InfectedSevereToInfectedCritical and "
258  "InfectedSevereToRecovered not equal to {:d}",
259  1);
260  return true;
261  }
262 
263  if (!floating_point_equal(
264  this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedCriticalToDead] +
265  this->get<TransitionProbabilities>()[(int)InfectionTransition::InfectedCriticalToRecovered],
266  1.0, 1e-14)) {
267  log_error("Constraint check: Sum of transition probability for InfectedCriticalToDead and "
268  "InfectedCriticalToRecovered not equal to {:d}",
269  1);
270  return true;
271  }
272 
273  for (size_t i = 0; i < (int)InfectionTransition::Count; i++) {
274  if (floating_point_less(this->get<TransitionDistributions>()[i].get_support_max(10), 0.0, 1e-14)) {
275  log_error("Constraint check: One parameter in TransitionDistributions has invalid support.");
276  return true;
277  }
278  }
279 
280  return false;
281  }
