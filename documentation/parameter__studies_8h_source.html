112  /*
113  * @brief Carry out all simulations in the parameter study.
114  * Save memory and enable more runs by immediately processing and/or discarding the result.
115  * The result processing function is called when a run is finished. It receives the result of the run
116  * (a SimulationGraph object) and an ordered index. The values returned by the result processing function
117  * are gathered and returned as a list.
118  * This function is parallelized if memilio is configured with MEMILIO_ENABLE_MPI.
119  * The MPI processes each contribute a share of the runs. The sample function and result processing function
120  * are called in the same process that performs the run. The results returned by the result processing function are
121  * gathered at the root process and returned as a list by the root in the same order as if the programm
122  * were running sequentially. Processes other than the root return an empty list.
123  * @param sample_graph Function that receives the ParametersGraph and returns a sampled copy.
124  * @param result_processing_function Processing function for simulation results, e.g., output function.
125  * @returns At the root process, a list of values per run that have been returned from the result processing function.
126  * At all other processes, an empty list.
127  * @tparam SampleGraphFunction Callable type, accepts instance of ParametersGraph.
128  * @tparam HandleSimulationResultFunction Callable type, accepts instance of SimulationGraph and an index of type size_t.
129  */
130  template <class SampleGraphFunction, class HandleSimulationResultFunction>
131  std::vector<std::invoke_result_t<HandleSimulationResultFunction, SimulationGraph, size_t>>
132  run(SampleGraphFunction sample_graph, HandleSimulationResultFunction result_processing_function)
133  {
134  int num_procs, rank;
135 #ifdef MEMILIO_ENABLE_MPI
136  MPI_Comm_size(mpi::get_world(), &num_procs);
137  MPI_Comm_rank(mpi::get_world(), &rank);
138 #else
139  num_procs = 1;
140  rank = 0;
141 #endif
142 
143  auto run_distribution = distribute_runs(m_num_runs, num_procs);
144  auto start_run_idx = std::accumulate(run_distribution.begin(), run_distribution.begin() + size_t(rank), size_t(0));
145  auto end_run_idx = start_run_idx + run_distribution[size_t(rank)];
146 
147  std::vector<std::invoke_result_t<HandleSimulationResultFunction, SimulationGraph, size_t>> ensemble_result;
148  ensemble_result.reserve(m_num_runs);
149 
150  for (size_t run_idx = start_run_idx; run_idx < end_run_idx; run_idx++) {
151  //ensure reproducible results if the number of runs or MPI process changes.
152  //assumptions:
153  //- the sample_graph functor uses the RNG provided by our library
154  //- the RNG has been freshly seeded/initialized before this call
155  //- the seeds are identical on all MPI processes
156  //- the block size of the RNG is sufficiently big to cover one run
157  // (when in doubt, use a larger block size; fast-forwarding the RNG is cheap and the period length
158  // of the mt19937 RNG is huge)
159  mio::thread_local_rng().forward_to_block(run_idx);
160 
161  auto sim = create_sampled_simulation(sample_graph);
162  sim.advance(m_tmax);
163 
164  ensemble_result.emplace_back(result_processing_function(std::move(sim).get_graph(), run_idx));
165  }
166 
167 #ifdef MEMILIO_ENABLE_MPI
168  //gather results
169  if (rank == 0) {
170  for (int src_rank = 1; src_rank < num_procs; ++src_rank) {
171  int bytes_size;
172  MPI_Recv(&bytes_size, 1, MPI_INT, src_rank, 0, mpi::get_world(), MPI_STATUS_IGNORE);
173  ByteStream bytes(bytes_size);
174  MPI_Recv(bytes.data(), bytes.data_size(), MPI_BYTE, src_rank, 0, mpi::get_world(), MPI_STATUS_IGNORE);
175 
176  auto src_ensemble_results = deserialize_binary(bytes, Tag<decltype(ensemble_result)>{});
177  if (!src_ensemble_results) {
178  log_error("Error receiving ensemble results from rank {}.", src_rank);
179  }
180  std::copy(src_ensemble_results.value().begin(), src_ensemble_results.value().end(),
181  std::back_inserter(ensemble_result));
182  }
183  }
184  else {
185  auto bytes = serialize_binary(ensemble_result);
186  auto bytes_size = int(bytes.data_size());
187  MPI_Send(&bytes_size, 1, MPI_INT, 0, 0, mpi::get_world());
188  MPI_Send(bytes.data(), bytes.data_size(), MPI_BYTE, 0, 0, mpi::get_world());
189  ensemble_result.clear(); //only return root process
190  }
191 #endif
192 
193  return ensemble_result;
194  }
195 
196  /*
197  * @brief Carry out all simulations in the parameter study.
198  * Convenience function for a few number of runs, but can use more memory because it stores all runs until the end.
199  * Unlike the other overload, this function is not MPI-parallel.
200  * @return vector of SimulationGraph for each run.
201  */
202  template <class SampleGraphFunction>
203  std::vector<SimulationGraph> run(SampleGraphFunction sample_graph)
204  {
205  std::vector<SimulationGraph> ensemble_result;
206  ensemble_result.reserve(m_num_runs);
207 
208  for (size_t i = 0; i < m_num_runs; i++) {
209  auto sim = create_sampled_simulation(sample_graph);
210  sim.advance(m_tmax);
211 
212  ensemble_result.emplace_back(std::move(sim).get_graph());
213  }
214 
215  return ensemble_result;
216  }

303  auto sampled_graph = sample_graph(m_graph);

314  std::vector<size_t> distribute_runs(size_t num_runs, int num_procs)
315  {
316  //evenly distribute runs
317  //lower processes do one more run if runs are not evenly distributable
318  auto num_runs_local = num_runs / num_procs; //integer division!
319  auto remainder = num_runs % num_procs;
320 
321  std::vector<size_t> run_distribution(num_procs);
322  std::fill(run_distribution.begin(), run_distribution.begin() + remainder, num_runs_local + 1);
323  std::fill(run_distribution.begin() + remainder, run_distribution.end(), num_runs_local);
324 
325  return run_distribution;
326  }
