488  template <class Iter>
489  static void inclusive_exclusive_sum_rec(Iter b, Iter e, Matrix& sum)
490  {
491  if (b != e) {
492  sum = sum + std::get<Matrix>(*b) - (sum.array() * std::get<Matrix>(*b).array()).matrix();
493  inclusive_exclusive_sum_rec(++b, e, sum);
494  }
495  }
496  template <class Tuple>
497  static Matrix inclusive_exclusive_sum(const std::vector<Tuple>& v)
498  {
499  assert(!v.empty());
500  auto& m = std::get<Matrix>(v.front());
501  auto sum = m.eval();
502  inclusive_exclusive_sum_rec(v.begin() + 1, v.end(), sum);
503  return sum;
504  }

512 template <class D>
513 void Dampings<D>::finalize() const
514 {
515  using std::get;
516 
517  if (m_accumulated_dampings_cached.empty()) {
518  m_accumulated_dampings_cached.emplace_back(Matrix::Zero(m_shape.rows(), m_shape.cols()),
519  SimulationTime(std::numeric_limits<double>::lowest()));
520 
521  std::vector<std::tuple<std::reference_wrapper<const Matrix>, DampingLevel, DampingType>> active_by_type;
522  std::vector<std::tuple<Matrix, DampingLevel>> sum_by_level;
523  for (auto& damping : m_dampings) {
524  //update active damping
525  update_active_dampings(damping, active_by_type, sum_by_level);
526  auto combined_damping = inclusive_exclusive_sum(sum_by_level);
527  assert((combined_damping.array() <= 1).all() && (combined_damping.array() >= 0).all() &&
528  "unexpected error, accumulated damping out of range.");
529  if (floating_point_equal(double(get<SimulationTime>(damping)),
530  double(get<SimulationTime>(m_accumulated_dampings_cached.back())), 1e-15, 1e-15)) {
531  std::get<Matrix>(m_accumulated_dampings_cached.back()) = combined_damping;
532  }
533  else {
534  m_accumulated_dampings_cached.emplace_back(combined_damping, get<SimulationTime>(damping));
535  }
536  }
537 
538  m_accumulated_dampings_cached.emplace_back(get<Matrix>(m_accumulated_dampings_cached.back()),
539  SimulationTime(std::numeric_limits<double>::max()));
540  }

543 template <class D>
544 void Dampings<D>::add_(const value_type& damping)
545 {
546  assert(damping.get_shape() == m_shape && "Inconsistent matrix shape.");
547  insert_sorted_replace(m_dampings, damping, [](auto& tup1, auto& tup2) {
548  return std::make_tuple(tup1.get_time(), int(tup1.get_type()), int(tup1.get_level())) <
549  std::make_tuple(tup2.get_time(), int(tup2.get_type()), int(tup2.get_level()));
550  });
551  m_accumulated_dampings_cached.clear();

554 template <class S>
555 void Dampings<S>::update_active_dampings(
556  const value_type& damping,
557  std::vector<std::tuple<std::reference_wrapper<const Matrix>, DampingLevel, DampingType>>& active_by_type,
558  std::vector<std::tuple<Matrix, DampingLevel>>& sum_by_level)
559 {
560  using std::get;
561 
562  const int MatrixIdx = 0;
563 
564  //find active with same type and level if existent
565  auto iter_active_same_type = std::find_if(active_by_type.begin(), active_by_type.end(), [&damping](auto& active) {
566  return get<DampingLevel>(active) == get<DampingLevel>(damping) &&
567  get<DampingType>(active) == get<DampingType>(damping);
568  });
569  if (iter_active_same_type != active_by_type.end()) {
570  //replace active of the same type and level
571  auto& active_same_type = *iter_active_same_type;
572  //find active with the same level
573  auto& sum_same_level = *std::find_if(sum_by_level.begin(), sum_by_level.end(), [&damping](auto& sum) {
574  return get<DampingLevel>(sum) == get<DampingLevel>(damping);
575  });
576  //remove active with the same type and level and add new one
577  get<MatrixIdx>(sum_same_level) += get<MatrixIdx>(damping) - get<MatrixIdx>(active_same_type).get();
578  //avoid negative values due to rounding error if e.g. a previous damping is lifted
579  get<MatrixIdx>(sum_same_level) = get<MatrixIdx>(sum_same_level).cwiseMax(0.);
580  get<MatrixIdx>(active_same_type) = get<MatrixIdx>(damping);
581  }
582  else {
583  //add new type to active
584  active_by_type.emplace_back(get<MatrixIdx>(damping), get<DampingLevel>(damping), get<DampingType>(damping));
585  //find damping with same level if existent
586  auto iter_sum_same_level = std::find_if(sum_by_level.begin(), sum_by_level.end(), [&damping](auto& sum) {
587  return get<DampingLevel>(sum) == get<DampingLevel>(damping);
588  });
589  if (iter_sum_same_level != sum_by_level.end()) {
590  //add to existing level
591  get<MatrixIdx>(*iter_sum_same_level) += get<MatrixIdx>(damping);
592  }
593  else {
594  //add new level
595  sum_by_level.emplace_back(get<MatrixIdx>(damping), get<DampingLevel>(damping));
596  }
597  }
598 }
