471 template <class IOContext, class F, class... T>
472 details::ApplyResultT<F, T...> apply(IOContext& io, F f, const IOResult<T>&... rs)
473 {
474  //store the status of each argument in an array to check them.
475  //it would be possible to write a recursive template that checks one argument
476  //after the other without copying each status. This would probably be slightly faster
477  //and easier to optimize for the compiler. but gdb crashes trying to resolve the
478  //very long symbol in case of many arguments. Successful results are very cheap to copy
479  //and slightly worse performance in the case of an error is probably acceptable.
480 
481  //check for errors in the arguments
482  IOStatus status[] = {(rs ? IOStatus{} : rs.error())...};
483  auto iter_err = std::find_if(std::begin(status), std::end(status), [](auto& s) {
484  return s.is_error();
485  });
486 
487  //evaluate f if all succesful
488  auto result =
489  iter_err == std::end(status) ? details::eval(f, rs...) : details::ApplyResultT<F, T...>(failure(*iter_err));
490 
491  if (!result) {
492  //save error in context
493  //if the error was generated by the applied function, the context hasn't seen it yet.
494  //this allows the context to fail fast and not continue to parse more.
495  io.set_error(result.error());
496  }
497  return result;
498 }

524 //recursive tuple deserialization for each tuple element
525 //read one tuple element after the other from the IOObject
526 //argument pack rs contains the elements that have been read already
527 template <class IOObj, class Tup, class... Ts>
528 std::enable_if_t<(sizeof...(Ts) == std::tuple_size<Tup>::value), IOResult<Tup>>
529 deserialize_tuple_element(IOObj& o, Tag<Tup>, const IOResult<Ts>&... rs)
530 {
531  //end of recursion
532  //number of arguments in rs is the same as the size of the tuple
533  //no more elements to read, so finalize the object
534  return mio::apply(
535  o,
536  [](const Ts&... ts) {
537  return Tup(ts...);
538  },
539  rs...);
540 }
541 template <class IOObj, class Tup, class... Ts>
542 std::enable_if_t<(sizeof...(Ts) < std::tuple_size<Tup>::value), IOResult<Tup>>
543 deserialize_tuple_element(IOObj& obj, Tag<Tup> tag, const IOResult<Ts>&... rs)
544 {
545  //get the next element of the tuple from the IO object
546  const size_t Idx = sizeof...(Ts);
547  auto r = obj.expect_element(make_tuple_element_name<Idx>(), Tag<std::tuple_element_t<Idx, Tup>>{});
548  //recurse, append the new element to the pack of arguments
549  return deserialize_tuple_element(obj, tag, rs..., r);
550 }

614 template <class IOContext, class M, std::enable_if_t<std::is_base_of<Eigen::EigenBase<M>, M>::value, void*> = nullptr>
615 IOResult<M> deserialize_internal(IOContext& io, Tag<M> /*tag*/)
616 {
617  auto obj = io.expect_object("Matrix");
618  auto rows = obj.expect_element("Rows", Tag<Eigen::Index>{});
619  auto cols = obj.expect_element("Columns", Tag<Eigen::Index>{});
620  auto elements = obj.expect_list("Elements", Tag<typename M::Scalar>{});
