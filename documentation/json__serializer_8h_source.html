457 inline IOResult<void> write_json(const std::string& path, const Json::Value& js_value)
458 {
459  std::ofstream ofs(path);
460  if (ofs.is_open()) {
461  Json::StreamWriterBuilder swb;
462  auto js_writer = std::unique_ptr<Json::StreamWriter>(swb.newStreamWriter());
463  js_writer->write(js_value, &ofs);
464  if (ofs) {
465  return success();
466  }
467  else {
468  return failure(StatusCode::UnknownError, "Unknown error writing json.");
469  }
470  }
471  else {
472  return failure(StatusCode::FileNotFound, path);
473  }

514 inline IOResult<Json::Value> read_json(const std::string& path)
515 {
516  std::ifstream ifs(path);
517  if (ifs.is_open()) {
518  Json::CharReaderBuilder crb;
519  std::string err_msg;
520  Json::Value js_value;
521  if (Json::parseFromStream(crb, ifs, &js_value, &err_msg)) {
522  return success(js_value);
523  }
524  return failure(StatusCode::UnknownError, path + ", " + err_msg);
525  }
526  return failure(StatusCode::FileNotFound, path);
