35 class Model : public CompartmentalModel<InfectionState, Populations<AgeGroup, InfectionState>, Parameters>
36 {
37  using Base = CompartmentalModel<InfectionState, mio::Populations<AgeGroup, InfectionState>, Parameters>;
38 
39 public:
40  Model(const Populations& pop, const ParameterSet& params)
41  : Base(pop, params)
42  {
43  }
44 
45  Model(int num_agegroups)
46  : Model(Populations({AgeGroup(num_agegroups), InfectionState::Count}), ParameterSet(AgeGroup(num_agegroups)))
47  {
48  }
49 
50  void get_derivatives(Eigen::Ref<const Eigen::VectorXd> pop, Eigen::Ref<const Eigen::VectorXd> y, double t,
51  Eigen::Ref<Eigen::VectorXd> dydt) const override
52  {
53  auto const& params = this->parameters;
54  AgeGroup n_agegroups = params.get_num_groups();
55 
56  ContactMatrixGroup const& contact_matrix = params.get<ContactPatterns>();
57 
58  auto icu_occupancy = 0.0;
59  auto test_and_trace_required = 0.0;
60  for (auto i = AgeGroup(0); i < n_agegroups; ++i) {
61  auto rateINS = 0.5 / (params.get<IncubationTime>()[i] - params.get<SerialInterval>()[i]);
62  test_and_trace_required +=
63  (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i]) * rateINS *
64  (this->populations.get_from(pop, {i, InfectionState::InfectedNoSymptomsNaive}) +
65  this->populations.get_from(pop, {i, InfectionState::InfectedNoSymptomsPartialImmunity}) +
66  this->populations.get_from(pop, {i, InfectionState::InfectedNoSymptomsImprovedImmunity}) +
67  this->populations.get_from(pop, {i, InfectionState::InfectedNoSymptomsNaiveConfirmed}) +
68  this->populations.get_from(pop, {i, InfectionState::InfectedNoSymptomsPartialImmunityConfirmed}) +
69  this->populations.get_from(pop, {i, InfectionState::InfectedNoSymptomsImprovedImmunityConfirmed}));
70  icu_occupancy += this->populations.get_from(pop, {i, InfectionState::InfectedCriticalNaive}) +
71  this->populations.get_from(pop, {i, InfectionState::InfectedCriticalPartialImmunity}) +
72  this->populations.get_from(pop, {i, InfectionState::InfectedCriticalImprovedImmunity});
73  }
74 
75  for (auto i = AgeGroup(0); i < n_agegroups; i++) {
76 
77  size_t SNi = this->populations.get_flat_index({i, InfectionState::SusceptibleNaive});
78  size_t ENi = this->populations.get_flat_index({i, InfectionState::ExposedNaive});
79  size_t INSNi = this->populations.get_flat_index({i, InfectionState::InfectedNoSymptomsNaive});
80  size_t ISyNi = this->populations.get_flat_index({i, InfectionState::InfectedSymptomsNaive});
81  size_t ISevNi = this->populations.get_flat_index({i, InfectionState::InfectedSevereNaive});
82  size_t ICrNi = this->populations.get_flat_index({i, InfectionState::InfectedCriticalNaive});
83 
84  size_t INSNCi = this->populations.get_flat_index({i, InfectionState::InfectedNoSymptomsNaiveConfirmed});
85  size_t ISyNCi = this->populations.get_flat_index({i, InfectionState::InfectedSymptomsNaiveConfirmed});
86 
87  size_t SPIi = this->populations.get_flat_index({i, InfectionState::SusceptiblePartialImmunity});
88  size_t EPIi = this->populations.get_flat_index({i, InfectionState::ExposedPartialImmunity});
89  size_t INSPIi = this->populations.get_flat_index({i, InfectionState::InfectedNoSymptomsPartialImmunity});
90  size_t ISyPIi = this->populations.get_flat_index({i, InfectionState::InfectedSymptomsPartialImmunity});
91  size_t ISevPIi = this->populations.get_flat_index({i, InfectionState::InfectedSeverePartialImmunity});
92  size_t ICrPIi = this->populations.get_flat_index({i, InfectionState::InfectedCriticalPartialImmunity});
93 
94  size_t INSPICi =
95  this->populations.get_flat_index({i, InfectionState::InfectedNoSymptomsPartialImmunityConfirmed});
96  size_t ISyPICi =
97  this->populations.get_flat_index({i, InfectionState::InfectedSymptomsPartialImmunityConfirmed});
98 
99  size_t EIIi = this->populations.get_flat_index({i, InfectionState::ExposedImprovedImmunity});
100  size_t INSIIi = this->populations.get_flat_index({i, InfectionState::InfectedNoSymptomsImprovedImmunity});
101  size_t ISyIIi = this->populations.get_flat_index({i, InfectionState::InfectedSymptomsImprovedImmunity});
102  size_t ISevIIi = this->populations.get_flat_index({i, InfectionState::InfectedSevereImprovedImmunity});
103  size_t ICrIIi = this->populations.get_flat_index({i, InfectionState::InfectedCriticalImprovedImmunity});
104 
105  size_t INSIICi =
106  this->populations.get_flat_index({i, InfectionState::InfectedNoSymptomsImprovedImmunityConfirmed});
107  size_t ISyIICi =
108  this->populations.get_flat_index({i, InfectionState::InfectedSymptomsImprovedImmunityConfirmed});
109 
110  size_t SIIi = this->populations.get_flat_index({i, InfectionState::SusceptibleImprovedImmunity});
111  size_t DNi = this->populations.get_flat_index({i, InfectionState::DeadNaive});
112  size_t DPIi = this->populations.get_flat_index({i, InfectionState::DeadPartialImmunity});
113  size_t DIIi = this->populations.get_flat_index({i, InfectionState::DeadImprovedImmunity});
114 
115  size_t ITi = this->populations.get_flat_index({i, InfectionState::TotalInfections});
116 
117  dydt[SNi] = 0;
118  dydt[ENi] = 0;
119 
120  dydt[SPIi] = 0;
121  dydt[EPIi] = 0;
122 
123  dydt[SIIi] = 0;
124  dydt[EIIi] = 0;
125 
126  double rateE = 1.0 / (2 * params.get<SerialInterval>()[i] - params.get<IncubationTime>()[i]);
127  double rateINS = 0.5 / (params.get<IncubationTime>()[i] - params.get<SerialInterval>()[i]);
128 
129  double reducExposedPartialImmunity = params.get<ReducExposedPartialImmunity>()[i];
130  double reducExposedImprovedImmunity = params.get<ReducExposedImprovedImmunity>()[i];
131  double reducInfectedSymptomsPartialImmunity = params.get<ReducInfectedSymptomsPartialImmunity>()[i];
132  double reducInfectedSymptomsImprovedImmunity = params.get<ReducInfectedSymptomsImprovedImmunity>()[i];
133  double reducInfectedSevereCriticalDeadPartialImmunity =
134  params.get<ReducInfectedSevereCriticalDeadPartialImmunity>()[i];
135  double reducInfectedSevereCriticalDeadImprovedImmunity =
136  params.get<ReducInfectedSevereCriticalDeadImprovedImmunity>()[i];
137  double reducTimeInfectedMild = params.get<ReducTimeInfectedMild>()[i];
138 
139  //symptomatic are less well quarantined when testing and tracing is overwhelmed so they infect more people
140  auto riskFromInfectedSymptomatic = smoother_cosine(
141  test_and_trace_required, params.get<TestAndTraceCapacity>(), params.get<TestAndTraceCapacity>() * 15,
142  params.get<RiskOfInfectionFromSymptomatic>()[i], params.get<MaxRiskOfInfectionFromSymptomatic>()[i]);
143 
144  auto riskFromInfectedNoSymptoms = smoother_cosine(
145  test_and_trace_required, params.get<TestAndTraceCapacity>(), params.get<TestAndTraceCapacity>() * 2,
146  params.get<RelativeTransmissionNoSymptoms>()[i], 1.0);
147 
148  for (auto j = AgeGroup(0); j < n_agegroups; j++) {
149  size_t SNj = this->populations.get_flat_index({j, InfectionState::SusceptibleNaive});
150  size_t ENj = this->populations.get_flat_index({j, InfectionState::ExposedNaive});
151  size_t INSNj = this->populations.get_flat_index({j, InfectionState::InfectedNoSymptomsNaive});
152  size_t ISyNj = this->populations.get_flat_index({j, InfectionState::InfectedSymptomsNaive});
153  size_t ISevNj = this->populations.get_flat_index({j, InfectionState::InfectedSevereNaive});
154  size_t ICrNj = this->populations.get_flat_index({j, InfectionState::InfectedCriticalNaive});
155  size_t SIIj = this->populations.get_flat_index({j, InfectionState::SusceptibleImprovedImmunity});
156 
157  size_t INSNCj = this->populations.get_flat_index({j, InfectionState::InfectedNoSymptomsNaiveConfirmed});
158  size_t ISyNCj = this->populations.get_flat_index({j, InfectionState::InfectedSymptomsNaiveConfirmed});
159 
160  size_t SPIj = this->populations.get_flat_index({j, InfectionState::SusceptiblePartialImmunity});
161  size_t EPIj = this->populations.get_flat_index({j, InfectionState::ExposedPartialImmunity});
162  size_t INSPIj =
163  this->populations.get_flat_index({j, InfectionState::InfectedNoSymptomsPartialImmunity});
164  size_t ISyPIj = this->populations.get_flat_index({j, InfectionState::InfectedSymptomsPartialImmunity});
165  size_t ISevPIj = this->populations.get_flat_index({j, InfectionState::InfectedSeverePartialImmunity});
166  size_t ICrPIj = this->populations.get_flat_index({j, InfectionState::InfectedCriticalPartialImmunity});
167 
168  size_t INSPICj =
169  this->populations.get_flat_index({j, InfectionState::InfectedNoSymptomsPartialImmunityConfirmed});
170  size_t ISyPICj =
171  this->populations.get_flat_index({j, InfectionState::InfectedSymptomsPartialImmunityConfirmed});
172 
173  size_t EIIj = this->populations.get_flat_index({j, InfectionState::ExposedImprovedImmunity});
174  size_t INSIIj =
175  this->populations.get_flat_index({j, InfectionState::InfectedNoSymptomsImprovedImmunity});
176  size_t ISyIIj = this->populations.get_flat_index({j, InfectionState::InfectedSymptomsImprovedImmunity});
177  size_t ISevIIj = this->populations.get_flat_index({j, InfectionState::InfectedSevereImprovedImmunity});
178  size_t ICrIIj = this->populations.get_flat_index({j, InfectionState::InfectedCriticalImprovedImmunity});
179 
180  size_t INSIICj =
181  this->populations.get_flat_index({j, InfectionState::InfectedNoSymptomsImprovedImmunityConfirmed});
182  size_t ISyIICj =
183  this->populations.get_flat_index({j, InfectionState::InfectedSymptomsImprovedImmunityConfirmed});
184 
185  // effective contact rate by contact rate between groups i and j and damping j
186  double season_val =
187  (1 + params.get<Seasonality>() *
188  sin(3.141592653589793 * (std::fmod((params.get<StartDay>() + t), 365.0) / 182.5 + 0.5)));
189  double cont_freq_eff =
190  season_val * contact_matrix.get_matrix_at(t)(static_cast<Eigen::Index>((size_t)i),
191  static_cast<Eigen::Index>((size_t)j));
192  // without died people
193  double Nj = pop[SNj] + pop[ENj] + pop[INSNj] + pop[ISyNj] + pop[ISevNj] + pop[ICrNj] + pop[INSNCj] +
194  pop[ISyNCj] + pop[SPIj] + pop[EPIj] + pop[INSPIj] + pop[ISyPIj] + pop[ISevPIj] +
195  pop[ICrPIj] + pop[INSPICj] + pop[ISyPICj] + pop[SIIj] + pop[EIIj] + pop[INSIIj] +
196  pop[ISyIIj] + pop[ISevIIj] + pop[ICrIIj] + pop[INSIICj] + pop[ISyIICj];
197 
198  double divNj = 1.0 / Nj; // precompute 1.0/Nj
199 
200  double ext_inf_force_dummy = cont_freq_eff * divNj *
201  params.template get<TransmissionProbabilityOnContact>()[(AgeGroup)i] *
202  (riskFromInfectedNoSymptoms * (pop[INSNj] + pop[INSPIj] + pop[INSIIj]) +
203  riskFromInfectedSymptomatic * (pop[ISyNj] + pop[ISyPIj] + pop[ISyIIj]));
204 
205  double dummy_SN = y[SNi] * ext_inf_force_dummy;
206 
207  double dummy_SPI = y[SPIi] * reducExposedPartialImmunity * ext_inf_force_dummy;
208 
209  double dummy_SII = y[SIIi] * reducExposedImprovedImmunity * ext_inf_force_dummy;
210 
211  dydt[SNi] -= dummy_SN;
212  dydt[ENi] += dummy_SN;
213 
214  dydt[SPIi] -= dummy_SPI;
215  dydt[EPIi] += dummy_SPI;
216 
217  dydt[SIIi] -= dummy_SII;
218  dydt[EIIi] += dummy_SII;
219  }
220 
221  // ICU capacity shortage is close
222  // TODO: if this is used with vaccination model, it has to be adapted if CriticalPerSevere
223  // is different for different vaccination status. This is not the case here and in addition, ICUCapacity
224  // is set to infinity and this functionality is deactivated, so this is OK for the moment.
225  double criticalPerSevereAdjusted =
226  smoother_cosine(icu_occupancy, 0.90 * params.get<ICUCapacity>(), params.get<ICUCapacity>(),
227  params.get<CriticalPerSevere>()[i], 0);
228 
229  double deathsPerSevereAdjusted = params.get<CriticalPerSevere>()[i] - criticalPerSevereAdjusted;
230 
231  /**** path of immune-naive ***/
232 
233  dydt[ENi] -= rateE * y[ENi]; // only exchange of E and InfectedNoSymptoms done here
234  dydt[INSNi] = rateE * y[ENi] - rateINS * y[INSNi];
235  dydt[INSNCi] = -rateINS * y[INSNCi];
236 
237  dydt[ISyNi] = (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i]) * rateINS * y[INSNi] -
238  (y[ISyNi] / params.get<TimeInfectedSymptoms>()[i]);
239  dydt[ISyNCi] = (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i]) * rateINS * y[INSNCi] -
240  (y[ISyNCi] / params.get<TimeInfectedSymptoms>()[i]);
241 
242  dydt[ISevNi] = params.get<SeverePerInfectedSymptoms>()[i] / params.get<TimeInfectedSymptoms>()[i] *
243  (y[ISyNi] + y[ISyNCi]) -
244  (1 / params.get<TimeInfectedSevere>()[i]) * y[ISevNi];
245  dydt[ICrNi] = -y[ICrNi] / params.get<TimeInfectedCritical>()[i];
246  // add flow from hosp to icu according to potentially adjusted probability due to ICU limits
247  dydt[ICrNi] += criticalPerSevereAdjusted / params.get<TimeInfectedSevere>()[i] * y[ISevNi];
248 
249  /**** path of partially immune (e.g., one dose of vaccination) ***/
250 
251  dydt[EPIi] -= rateE * y[EPIi]; // only exchange of E and InfectedNoSymptoms done here
252  dydt[INSPIi] = rateE * y[EPIi] - (rateINS / reducTimeInfectedMild) * y[INSPIi];
253  dydt[INSPICi] = -(rateINS / reducTimeInfectedMild) * y[INSPICi];
254  dydt[ISyPIi] = (reducInfectedSymptomsPartialImmunity / reducExposedPartialImmunity) *
255  (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i]) *
256  (rateINS / reducTimeInfectedMild) * y[INSPIi] -
257  (y[ISyPIi] / (params.get<TimeInfectedSymptoms>()[i] * reducTimeInfectedMild));
258  dydt[ISyPICi] = (reducInfectedSymptomsPartialImmunity / reducExposedPartialImmunity) *
259  (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i]) *
260  (rateINS / reducTimeInfectedMild) * y[INSPICi] -
261  (y[ISyPICi] / (params.get<TimeInfectedSymptoms>()[i] * reducTimeInfectedMild));
262  dydt[ISevPIi] = reducInfectedSevereCriticalDeadPartialImmunity / reducInfectedSymptomsPartialImmunity *
263  params.get<SeverePerInfectedSymptoms>()[i] /
264  (params.get<TimeInfectedSymptoms>()[i] * reducTimeInfectedMild) *
265  (y[ISyPIi] + y[ISyPICi]) -
266  (1 / params.get<TimeInfectedSevere>()[i]) * y[ISevPIi];
267  dydt[ICrPIi] = -(1 / params.get<TimeInfectedCritical>()[i]) * y[ICrPIi];
268  // add flow from hosp to icu according to potentially adjusted probability due to ICU limits
269  dydt[ICrPIi] += reducInfectedSevereCriticalDeadPartialImmunity /
270  reducInfectedSevereCriticalDeadPartialImmunity * criticalPerSevereAdjusted /
271  params.get<TimeInfectedSevere>()[i] * y[ISevPIi];
272 
273  /**** path of twice vaccinated, here called immune although reinfection is possible now ***/
274 
275  dydt[EIIi] -= rateE * y[EIIi]; // only exchange of E and C done here
276 
277  dydt[INSIIi] = rateE * y[EIIi] - (rateINS / reducTimeInfectedMild) * y[INSIIi];
278  dydt[INSIICi] = -(rateINS / reducTimeInfectedMild) * y[INSIICi];
279 
280  dydt[ISyIIi] = (reducInfectedSymptomsImprovedImmunity / reducExposedImprovedImmunity) *
281  (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i]) *
282  (rateINS / reducTimeInfectedMild) * y[INSIIi] -
283  (1 / (params.get<TimeInfectedSymptoms>()[i] * reducTimeInfectedMild)) * y[ISyIIi];
284  dydt[ISyIICi] = (reducInfectedSymptomsImprovedImmunity / reducExposedImprovedImmunity) *
285  (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i]) *
286  (rateINS / reducTimeInfectedMild) * y[INSIICi] -
287  (1 / (params.get<TimeInfectedSymptoms>()[i] * reducTimeInfectedMild)) * y[ISyIICi];
288  dydt[ISevIIi] = reducInfectedSevereCriticalDeadImprovedImmunity / reducInfectedSymptomsImprovedImmunity *
289  params.get<SeverePerInfectedSymptoms>()[i] /
290  (params.get<TimeInfectedSymptoms>()[i] * reducTimeInfectedMild) *
291  (y[ISyIIi] + y[ISyIICi]) -
292  (1 / params.get<TimeInfectedSevere>()[i]) * y[ISevIIi];
293  dydt[ICrIIi] = -(1 / params.get<TimeInfectedCritical>()[i]) * y[ICrIIi];
294  // add flow from hosp to icu according to potentially adjusted probability due to ICU limits
295  dydt[ICrIIi] += reducInfectedSevereCriticalDeadImprovedImmunity /
296  reducInfectedSevereCriticalDeadImprovedImmunity * criticalPerSevereAdjusted /
297  params.get<TimeInfectedSevere>()[i] * y[ISevIIi];
298 
299  // compute auxiliary compartment of all past infections
300  dydt[ITi] =
301  (1 / params.get<TimeInfectedSymptoms>()[i]) * (y[ISyNi] + y[ISyNCi]) +
302  (1 / (params.get<TimeInfectedSymptoms>()[i] * reducTimeInfectedMild)) * (y[ISyPIi] + y[ISyPICi]) +
303  (1 / (params.get<TimeInfectedSymptoms>()[i] * reducTimeInfectedMild)) * (y[ISyIIi] + y[ISyIICi]);
304 
305  // recovered and deaths from all paths
306  dydt[SIIi] += params.get<RecoveredPerInfectedNoSymptoms>()[i] * rateINS * (y[INSNi] + y[INSNCi]) +
307  (1 - params.get<SeverePerInfectedSymptoms>()[i]) / params.get<TimeInfectedSymptoms>()[i] *
308  (y[ISyNi] + y[ISyNCi]) +
309  (1 - params.get<CriticalPerSevere>()[i]) / params.get<TimeInfectedSevere>()[i] * y[ISevNi] +
310  (1 - params.get<DeathsPerCritical>()[i]) / params.get<TimeInfectedCritical>()[i] * y[ICrNi];
311 
312  dydt[SIIi] +=
313  (1 - (reducInfectedSymptomsPartialImmunity / reducExposedPartialImmunity) *
314  (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i])) *
315  rateINS / reducTimeInfectedMild * (y[INSPIi] + y[INSPICi]) +
316  (1 - (reducInfectedSevereCriticalDeadPartialImmunity / reducInfectedSymptomsPartialImmunity) *
317  params.get<SeverePerInfectedSymptoms>()[i]) /
318  (params.get<TimeInfectedSymptoms>()[i] * reducTimeInfectedMild) * (y[ISyPIi] + y[ISyPICi]) +
319  (1 - (reducInfectedSevereCriticalDeadPartialImmunity / reducInfectedSevereCriticalDeadPartialImmunity) *
320  params.get<CriticalPerSevere>()[i]) /
321  params.get<TimeInfectedSevere>()[i] * y[ISevPIi] +
322  (1 - (reducInfectedSevereCriticalDeadPartialImmunity / reducInfectedSevereCriticalDeadPartialImmunity) *
323  params.get<DeathsPerCritical>()[i]) /
324  params.get<TimeInfectedCritical>()[i] * y[ICrPIi];
325 
326  dydt[SIIi] +=
327  (1 - (reducInfectedSymptomsImprovedImmunity / reducExposedImprovedImmunity) *
328  (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i])) *
329  rateINS / reducTimeInfectedMild * (y[INSIIi] + y[INSIICi]) +
330  (1 - (reducInfectedSevereCriticalDeadImprovedImmunity / reducInfectedSymptomsImprovedImmunity) *
331  params.get<SeverePerInfectedSymptoms>()[i]) /
332  (params.get<TimeInfectedSymptoms>()[i] * reducTimeInfectedMild) * (y[ISyIIi] + y[ISyIICi]) +
333  (1 -
334  (reducInfectedSevereCriticalDeadImprovedImmunity / reducInfectedSevereCriticalDeadImprovedImmunity) *
335  params.get<CriticalPerSevere>()[i]) /
336  params.get<TimeInfectedSevere>()[i] * y[ISevIIi] +
337  (1 -
338  (reducInfectedSevereCriticalDeadImprovedImmunity / reducInfectedSevereCriticalDeadImprovedImmunity) *
339  params.get<DeathsPerCritical>()[i]) /
340  params.get<TimeInfectedCritical>()[i] * y[ICrIIi];
341 
342  dydt[DNi] = params.get<DeathsPerCritical>()[i] / params.get<TimeInfectedCritical>()[i] * y[ICrNi];
343  dydt[DPIi] = reducInfectedSevereCriticalDeadPartialImmunity /
344  reducInfectedSevereCriticalDeadPartialImmunity * params.get<DeathsPerCritical>()[i] /
345  params.get<TimeInfectedCritical>()[i] * y[ICrPIi];
346  dydt[DIIi] = reducInfectedSevereCriticalDeadImprovedImmunity /
347  reducInfectedSevereCriticalDeadImprovedImmunity * params.get<DeathsPerCritical>()[i] /
348  params.get<TimeInfectedCritical>()[i] * y[ICrIIi];
349  // add potential, additional deaths due to ICU overflow
350  dydt[DNi] += deathsPerSevereAdjusted / params.get<TimeInfectedSevere>()[i] * y[ISevNi];
351  dydt[DPIi] +=
352  (reducInfectedSevereCriticalDeadPartialImmunity / reducInfectedSevereCriticalDeadPartialImmunity) *
353  deathsPerSevereAdjusted / params.get<TimeInfectedSevere>()[i] * y[ISevPIi];
354  dydt[DIIi] +=
355  (reducInfectedSevereCriticalDeadImprovedImmunity / reducInfectedSevereCriticalDeadImprovedImmunity) *
356  deathsPerSevereAdjusted / params.get<TimeInfectedSevere>()[i] * y[ISevIIi];
357  }
358  }

376  template <class IOContext>
377  static IOResult<Model> deserialize(IOContext& io)
378  {
379  auto obj = io.expect_object("Model");
380  auto par = obj.expect_element("Parameters", Tag<ParameterSet>{});
381  auto pop = obj.expect_element("Populations", Tag<Populations>{});
382  return apply(
383  io,
384  [](auto&& par_, auto&& pop_) {
385  return Model{pop_, par_};
386  },

425  void apply_b161(double t)
426  {
427 
428  auto start_day = this->get_model().parameters.template get<StartDay>();
429  auto b161_growth = (start_day - get_day_in_year(Date(2021, 6, 6))) * 0.1666667;
430  // 2 equal to the share of the delta variant on June 6
431  double share_new_variant = std::min(1.0, pow(2, t * 0.1666667 + b161_growth) * 0.01);
432  size_t num_groups = (size_t)this->get_model().parameters.get_num_groups();
433  for (size_t i = 0; i < num_groups; ++i) {
434  double new_transmission =
435  (1 - share_new_variant) *
436  this->get_model().parameters.template get<BaseInfectiousnessB117>()[(AgeGroup)i] +
437  share_new_variant * this->get_model().parameters.template get<BaseInfectiousnessB161>()[(AgeGroup)i];
438  this->get_model().parameters.template get<TransmissionProbabilityOnContact>()[(AgeGroup)i] =
439  new_transmission;
440  }
441  }

443  void apply_vaccination(double t)
444  {
445  auto t_idx = SimulationDay((size_t)t);
446  auto& params = this->get_model().parameters;
447  size_t num_groups = (size_t)params.get_num_groups();
448  auto last_value = this->get_result().get_last_value();
449 
450  auto count = (size_t)InfectionState::Count;
451  auto S = (size_t)InfectionState::SusceptibleNaive;
452  auto SV = (size_t)InfectionState::SusceptiblePartialImmunity;
453  auto R = (size_t)InfectionState::SusceptibleImprovedImmunity;
454 
455  for (size_t i = 0; i < num_groups; ++i) {
456 
457  double first_vacc;
458  double full_vacc;
459  if (t_idx == SimulationDay(0)) {
460  first_vacc = params.template get<DailyFirstVaccination>()[{(AgeGroup)i, t_idx}];
461  full_vacc = params.template get<DailyFullVaccination>()[{(AgeGroup)i, t_idx}];
462  }
463  else {
464  first_vacc = params.template get<DailyFirstVaccination>()[{(AgeGroup)i, t_idx}] -
465  params.template get<DailyFirstVaccination>()[{(AgeGroup)i, t_idx - SimulationDay(1)}];
466  full_vacc = params.template get<DailyFullVaccination>()[{(AgeGroup)i, t_idx}] -
467  params.template get<DailyFullVaccination>()[{(AgeGroup)i, t_idx - SimulationDay(1)}];
468  }
469 
470  if (last_value(count * i + S) - first_vacc < 0) {
471  auto corrected = 0.99 * last_value(count * i + S);
472  log_warning("too many first vaccinated at time {}: setting first_vacc from {} to {}", t, first_vacc,
473  corrected);
474  first_vacc = corrected;
475  }
476 
477  last_value(count * i + S) -= first_vacc;
478  last_value(count * i + SV) += first_vacc;
479 
480  if (last_value(count * i + SV) - full_vacc < 0) {
481  auto corrected = 0.99 * last_value(count * i + SV);
482  log_warning("too many fully vaccinated at time {}: setting full_vacc from {} to {}", t, full_vacc,
483  corrected);
484  full_vacc = corrected;
485  }
486 
487  last_value(count * i + SV) -= full_vacc;
488  last_value(count * i + R) += full_vacc;
489  }
490  }

499  Eigen::Ref<Eigen::VectorXd> advance(double tmax)
500  {
501  auto& t_end_dyn_npis = this->get_model().parameters.get_end_dynamic_npis();
502  auto& dyn_npis = this->get_model().parameters.template get<DynamicNPIsInfectedSymptoms>();
503  auto& contact_patterns = this->get_model().parameters.template get<ContactPatterns>();
504 
505  double delay_lockdown;
506  auto t = Base::get_result().get_last_time();
507  const auto dt = dyn_npis.get_interval().get();
508  while (t < tmax) {
509 
510  auto dt_eff = std::min({dt, tmax - t, m_t_last_npi_check + dt - t});
511  if (dt_eff >= 1.0) {
512  dt_eff = 1.0;
513  }
514 
515  if (t == 0) {
516  //this->apply_vaccination(t); // done in init now?
517  this->apply_b161(t);
518  }
519  Base::advance(t + dt_eff);
520  if (t + 0.5 + dt_eff - std::floor(t + 0.5) >= 1) {
521  this->apply_vaccination(t + 0.5 + dt_eff);
522  this->apply_b161(t);
523  }
524 
525  if (t > 0) {
526  delay_lockdown = 7;
527  }
528  else {
529  delay_lockdown = 0;
530  }
531  t = t + dt_eff;
532 
533  if (dyn_npis.get_thresholds().size() > 0) {
534  if (floating_point_greater_equal(t, m_t_last_npi_check + dt)) {
535  if (t < t_end_dyn_npis) {
536  auto inf_rel = get_infections_relative(*this, t, this->get_result().get_last_value()) *
537  dyn_npis.get_base_value();
538  auto exceeded_threshold = dyn_npis.get_max_exceeded_threshold(inf_rel);
539  if (exceeded_threshold != dyn_npis.get_thresholds().end() &&
540  (exceeded_threshold->first > m_dynamic_npi.first ||
541  t > double(m_dynamic_npi.second))) { //old npi was weaker or is expired
542 
543  auto t_start = SimulationTime(t + delay_lockdown);
544  auto t_end = t_start + SimulationTime(dyn_npis.get_duration());
545  this->get_model().parameters.get_start_commuter_detection() = (double)t_start;
546  this->get_model().parameters.get_end_commuter_detection() = (double)t_end;
547  m_dynamic_npi = std::make_pair(exceeded_threshold->first, t_end);
548  implement_dynamic_npis(contact_patterns.get_cont_freq_mat(), exceeded_threshold->second,
549  t_start, t_end, [](auto& g) {
550  return make_contact_damping_matrix(g);
551  });
552  }
553  }
554  m_t_last_npi_check = t;
555  }
556  }
557  else {
558  m_t_last_npi_check = t;
559  }
560  }
561  return this->get_result().get_last_value();
562  }
563 
564 private:
565  double m_t_last_npi_check;
566  std::pair<double, SimulationTime> m_dynamic_npi = {-std::numeric_limits<double>::max(), SimulationTime(0)};
567 };

613 template <class Base = mio::Simulation<Model>>
614 auto get_migration_factors(const Simulation<Base>& sim, double /*t*/, const Eigen::Ref<const Eigen::VectorXd>& y)
615 {
616  auto& params = sim.get_model().parameters;
617  //parameters as arrays
618  auto& t_inc = params.template get<IncubationTime>().array().template cast<double>();
619  auto& t_ser = params.template get<SerialInterval>().array().template cast<double>();
620  auto& p_asymp = params.template get<RecoveredPerInfectedNoSymptoms>().array().template cast<double>();
621  auto& p_inf = params.template get<RiskOfInfectionFromSymptomatic>().array().template cast<double>();
622  auto& p_inf_max = params.template get<MaxRiskOfInfectionFromSymptomatic>().array().template cast<double>();
623  //slice of InfectedNoSymptoms
624  auto y_car = slice(y, {Eigen::Index(InfectionState::InfectedNoSymptomsNaive),
625  Eigen::Index(size_t(params.get_num_groups())), Eigen::Index(InfectionState::Count)}) +
626  slice(y, {Eigen::Index(InfectionState::InfectedNoSymptomsPartialImmunity),
627  Eigen::Index(size_t(params.get_num_groups())), Eigen::Index(InfectionState::Count)}) +
628  slice(y, {Eigen::Index(InfectionState::InfectedNoSymptomsImprovedImmunity),
629  Eigen::Index(size_t(params.get_num_groups())), Eigen::Index(InfectionState::Count)});
630 
631  //compute isolation, same as infection risk from main model
632  auto R3 = 0.5 / (t_inc - t_ser);
633  auto test_and_trace_required = ((1 - p_asymp) * R3 * y_car.array()).sum();
634  auto riskFromInfectedSymptomatic =
635  smoother_cosine(test_and_trace_required, double(params.template get<TestAndTraceCapacity>()),
636  params.template get<TestAndTraceCapacity>() * 5, p_inf.matrix(), p_inf_max.matrix());
637 
638  //set factor for infected
639  auto factors = Eigen::VectorXd::Ones(y.rows()).eval();
640  slice(factors, {Eigen::Index(InfectionState::InfectedSymptomsNaive), Eigen::Index(size_t(params.get_num_groups())),
641  Eigen::Index(InfectionState::Count)})
642  .array() = riskFromInfectedSymptomatic;
643  slice(factors, {Eigen::Index(InfectionState::InfectedSymptomsPartialImmunity),
644  Eigen::Index(size_t(params.get_num_groups())), Eigen::Index(InfectionState::Count)})
645  .array() = riskFromInfectedSymptomatic;
646  slice(factors, {Eigen::Index(InfectionState::InfectedSymptomsImprovedImmunity),
647  Eigen::Index(size_t(params.get_num_groups())), Eigen::Index(InfectionState::Count)})
648  .array() = riskFromInfectedSymptomatic;
649  return factors;
650 }

653 auto test_commuters(Simulation<Base>& sim, Eigen::Ref<Eigen::VectorXd> migrated, double time)
654 {

657  if (time >= model.parameters.get_start_commuter_detection() &&
658  time < model.parameters.get_end_commuter_detection()) {

660  }

662  auto ISyNi = model.populations.get_flat_index({i, InfectionState::InfectedSymptomsNaive});
663  auto ISyNCi = model.populations.get_flat_index({i, InfectionState::InfectedSymptomsNaiveConfirmed});
664  auto INSNi = model.populations.get_flat_index({i, InfectionState::InfectedNoSymptomsNaive});
665  auto INSNCi = model.populations.get_flat_index({i, InfectionState::InfectedNoSymptomsNaiveConfirmed});
666 
667  auto ISPIi = model.populations.get_flat_index({i, InfectionState::InfectedSymptomsPartialImmunity});
668  auto ISPICi = model.populations.get_flat_index({i, InfectionState::InfectedSymptomsPartialImmunityConfirmed});
669  auto INSPIi = model.populations.get_flat_index({i, InfectionState::InfectedNoSymptomsPartialImmunity});
670  auto INSPICi =
671  model.populations.get_flat_index({i, InfectionState::InfectedNoSymptomsPartialImmunityConfirmed});
672 
673  auto ISyIIi = model.populations.get_flat_index({i, InfectionState::InfectedSymptomsImprovedImmunity});
674  auto ISyIICi = model.populations.get_flat_index({i, InfectionState::InfectedSymptomsImprovedImmunityConfirmed});
675  auto INSIIi = model.populations.get_flat_index({i, InfectionState::InfectedNoSymptomsImprovedImmunity});
676  auto INSIICi =
677  model.populations.get_flat_index({i, InfectionState::InfectedNoSymptomsImprovedImmunityConfirmed});
678 
679  //put detected commuters in their own compartment so they don't contribute to infections in their home node
680  sim.get_result().get_last_value()[ISyNi] -= migrated[ISyNi] * (1 - nondetection);
681  sim.get_result().get_last_value()[ISyNCi] += migrated[ISyNi] * (1 - nondetection);

687  sim.get_result().get_last_value()[INSPIi] -= migrated[INSPIi] * (1 - nondetection);
688  sim.get_result().get_last_value()[INSPICi] += migrated[INSPIi] * (1 - nondetection);

692  sim.get_result().get_last_value()[INSIIi] -= migrated[INSIIi] * (1 - nondetection);
693  sim.get_result().get_last_value()[INSIICi] += migrated[INSIIi] * (1 - nondetection);
694 
695  //reduce the number of commuters
696  migrated[ISyNi] *= nondetection;
697  migrated[INSNi] *= nondetection;
698 
699  migrated[ISPIi] *= nondetection;
