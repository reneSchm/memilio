48  void get_derivatives(Eigen::Ref<const Eigen::VectorXd> pop, Eigen::Ref<const Eigen::VectorXd> y, double t,
49  Eigen::Ref<Eigen::VectorXd> dydt) const override
50  {
51  auto& params = this->parameters;
52  double coeffStoE = params.get<ContactPatterns>().get_matrix_at(t)(0, 0) *
53  params.get<TransmissionProbabilityOnContact>() / populations.get_total();
54 
55  dydt[(size_t)InfectionState::Susceptible] =
56  -coeffStoE * y[(size_t)InfectionState::Susceptible] * pop[(size_t)InfectionState::Infected];
57  dydt[(size_t)InfectionState::Exposed] =
58  coeffStoE * y[(size_t)InfectionState::Susceptible] * pop[(size_t)InfectionState::Infected] -
59  (1.0 / params.get<TimeExposed>()) * y[(size_t)InfectionState::Exposed];
60  dydt[(size_t)InfectionState::Infected] =
61  (1.0 / params.get<TimeExposed>()) * y[(size_t)InfectionState::Exposed] -
62  (1.0 / params.get<TimeInfected>()) * y[(size_t)InfectionState::Infected];
63  dydt[(size_t)InfectionState::Recovered] =
64  (1.0 / params.get<TimeInfected>()) * y[(size_t)InfectionState::Infected];
65  }
66 };
