48 T random_transition(T current_state, TimeSpan dt, const std::pair<T, double> (&transitions)[NumTransitions])
49 {
50  assert(std::all_of(std::begin(transitions), std::end(transitions),
51  [](auto& p) {
52  return p.second >= 0.0;
53  }) &&
54  "transition rates must be non-negative");
55 
56  //check if any transition happens using exponential distribution with the sum of all transition rates
57  auto sum = std::accumulate(std::begin(transitions), std::end(transitions), 0.0, [](auto&& a, auto&& t) {
58  return a + t.second;
59  });
60  if (sum <= 0) { //no transitions or all transitions have rate zero
61  return current_state;
62  }
63  auto v = ExponentialDistribution<double>::get_instance()(sum);
64  if (v < dt.days()) {
65  //pick one of the possible transitions using discrete distribution
66  std::array<double, NumTransitions> rates;
67  std::transform(std::begin(transitions), std::end(transitions), rates.begin(), [](auto&& t) {
68  return t.second;
69  });
70  auto random_idx = DiscreteDistribution<size_t>::get_instance()(rates);
71  return transitions[random_idx].first;
72  }
73 
74  return current_state;
75 }
