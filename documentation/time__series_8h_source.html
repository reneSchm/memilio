502 inline Eigen::Index next_pow2(Eigen::Index i)
503 {
504  //https://stackoverflow.com/questions/1322510/given-an-integer-how-do-i-find-the-next-largest-power-of-two-using-bit-twiddlin
505  --i;
506  i |= i >> 1;
507  i |= i >> 2;
508  i |= i >> 4;
509  i |= i >> 8;
510  i |= i >> 16;
511  if constexpr (sizeof(Eigen::Index) == 8)
512  {
513  i |= i >> 32;
514  }
515  ++i;
516  return i;
517 }

765 decltype(std::declval<TS>().rend()) find_value_reverse(TS&& ts, FP t_search, FP abs_tol = 0, FP rel_tol = 0)
766 {
767  auto iter_t = find_if(ts.get_reverse_times().begin(), ts.get_reverse_times().end(), [=](auto t) {
