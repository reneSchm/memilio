56  void get_derivatives(Eigen::Ref<const Eigen::VectorXd> pop, Eigen::Ref<const Eigen::VectorXd> y, double t,
57  Eigen::Ref<Eigen::VectorXd> dydt) const override
58  {
59  // alpha // percentage of asymptomatic cases
60  // beta // risk of infection from the infected symptomatic patients
61  // rho // hospitalized per infectious
62  // theta // icu per hospitalized
63  // delta // deaths per ICUs
64  // 0: S, 1: E, 2: C, 3: I, 4: H, 5: U, 6: R, 7: D
65  auto const& params = this->parameters;
66  AgeGroup n_agegroups = params.get_num_groups();
67 
68  ContactMatrixGroup const& contact_matrix = params.get<ContactPatterns>();
69 
70  auto icu_occupancy = 0.0;
71  auto test_and_trace_required = 0.0;
72  for (auto i = AgeGroup(0); i < n_agegroups; ++i) {
73  auto rateINS = 0.5 / (params.get<IncubationTime>()[i] - params.get<SerialInterval>()[i]);
74  test_and_trace_required += (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i]) * rateINS *
75  this->populations.get_from(pop, {i, InfectionState::InfectedNoSymptoms});
76  icu_occupancy += this->populations.get_from(pop, {i, InfectionState::InfectedCritical});
77  }
78 
79  for (auto i = AgeGroup(0); i < n_agegroups; i++) {
80 
81  size_t Si = this->populations.get_flat_index({i, InfectionState::Susceptible});
82  size_t Ei = this->populations.get_flat_index({i, InfectionState::Exposed});
83  size_t INSi = this->populations.get_flat_index({i, InfectionState::InfectedNoSymptoms});
84  size_t INSCi = this->populations.get_flat_index({i, InfectionState::InfectedNoSymptomsConfirmed});
85  size_t ISyi = this->populations.get_flat_index({i, InfectionState::InfectedSymptoms});
86  size_t ISyCi = this->populations.get_flat_index({i, InfectionState::InfectedSymptomsConfirmed});
87  size_t ISevi = this->populations.get_flat_index({i, InfectionState::InfectedSevere});
88  size_t ICri = this->populations.get_flat_index({i, InfectionState::InfectedCritical});
89  size_t Ri = this->populations.get_flat_index({i, InfectionState::Recovered});
90  size_t Di = this->populations.get_flat_index({i, InfectionState::Dead});
91 
92  dydt[Si] = 0;
93  dydt[Ei] = 0;
94 
95  double rateE =
96  1.0 / (2 * params.get<SerialInterval>()[i] - params.get<IncubationTime>()[i]); // R2 = 1/(2SI-TINC)
97  double rateINS =
98  0.5 / (params.get<IncubationTime>()[i] - params.get<SerialInterval>()[i]); // R3 = 1/(2(TINC-SI))
99 
100  for (auto j = AgeGroup(0); j < n_agegroups; j++) {
101  size_t Sj = this->populations.get_flat_index({j, InfectionState::Susceptible});
102  size_t Ej = this->populations.get_flat_index({j, InfectionState::Exposed});
103  size_t INSj = this->populations.get_flat_index({j, InfectionState::InfectedNoSymptoms});
104  size_t ISyj = this->populations.get_flat_index({j, InfectionState::InfectedSymptoms});
105  size_t ISevj = this->populations.get_flat_index({j, InfectionState::InfectedSevere});
106  size_t ICrj = this->populations.get_flat_index({j, InfectionState::InfectedCritical});
107  size_t Rj = this->populations.get_flat_index({j, InfectionState::Recovered});
108 
109  //symptomatic are less well quarantined when testing and tracing is overwhelmed so they infect more people
110  auto riskFromInfectedSymptomatic = smoother_cosine(
111  test_and_trace_required, params.get<TestAndTraceCapacity>(), params.get<TestAndTraceCapacity>() * 5,
112  params.get<RiskOfInfectionFromSymptomatic>()[j],
113  params.get<MaxRiskOfInfectionFromSymptomatic>()[j]);
114 
115  // effective contact rate by contact rate between groups i and j and damping j
116  double season_val =
117  (1 + params.get<Seasonality>() *
118  sin(3.141592653589793 * (std::fmod((params.get<StartDay>() + t), 365.0) / 182.5 + 0.5)));
119  double cont_freq_eff =
120  season_val * contact_matrix.get_matrix_at(t)(static_cast<Eigen::Index>((size_t)i),
121  static_cast<Eigen::Index>((size_t)j));
122  double Nj =
123  pop[Sj] + pop[Ej] + pop[INSj] + pop[ISyj] + pop[ISevj] + pop[ICrj] + pop[Rj]; // without died people
124  double divNj = 1.0 / Nj; // precompute 1.0/Nj
125  double dummy_S = y[Si] * cont_freq_eff * divNj * params.get<TransmissionProbabilityOnContact>()[i] *
126  (params.get<RelativeTransmissionNoSymptoms>()[j] * pop[INSj] +
127  riskFromInfectedSymptomatic * pop[ISyj]);
128 
129  dydt[Si] -= dummy_S;
130  dydt[Ei] += dummy_S;
131  }
132 
133  // ICU capacity shortage is close
134  double criticalPerSevereAdjusted =
135  smoother_cosine(icu_occupancy, 0.90 * params.get<ICUCapacity>(), params.get<ICUCapacity>(),
136  params.get<CriticalPerSevere>()[i], 0);
137 
138  double deathsPerSevereAdjusted = params.get<CriticalPerSevere>()[i] - criticalPerSevereAdjusted;
139 
140  dydt[Ei] -= rateE * y[Ei]; // only exchange of E and INS done here
141  dydt[INSi] = rateE * y[Ei] - rateINS * y[INSi];
142  dydt[INSCi] = -rateINS * y[INSCi];
143  dydt[ISyi] = (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i]) * rateINS * y[INSi] -
144  (1 / params.get<TimeInfectedSymptoms>()[i]) * y[ISyi];
145  dydt[ISyCi] = (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i]) * rateINS * y[INSCi] -
146  (1 / params.get<TimeInfectedSymptoms>()[i]) * y[ISyCi];
147  dydt[ISevi] = params.get<SeverePerInfectedSymptoms>()[i] / params.get<TimeInfectedSymptoms>()[i] * y[ISyi] -
148  (1 / params.get<TimeInfectedSevere>()[i]) * y[ISevi];
149  dydt[ICri] = -(1 / params.get<TimeInfectedCritical>()[i]) * y[ICri];
150  // add flow from hosp to icu according to potentially adjusted probability due to ICU limits
151  dydt[ICri] += criticalPerSevereAdjusted / params.get<TimeInfectedSevere>()[i] * y[ISevi];
152 
153  dydt[Ri] = params.get<RecoveredPerInfectedNoSymptoms>()[i] * rateINS * (y[INSi] + y[INSCi]) +
154  (1 - params.get<SeverePerInfectedSymptoms>()[i]) / params.get<TimeInfectedSymptoms>()[i] *
155  (y[ISyi] + y[ISyCi]) +
156  (1 - params.get<CriticalPerSevere>()[i]) / params.get<TimeInfectedSevere>()[i] * y[ISevi] +
157  (1 - params.get<DeathsPerCritical>()[i]) / params.get<TimeInfectedCritical>()[i] * y[ICri];
158 
159  dydt[Di] = params.get<DeathsPerCritical>()[i] / params.get<TimeInfectedCritical>()[i] * y[ICri];
160  // add potential, additional deaths due to ICU overflow
161  dydt[Di] += deathsPerSevereAdjusted / params.get<TimeInfectedSevere>()[i] * y[ISevi];
162  }
163  }

239  Eigen::Ref<Eigen::VectorXd> advance(double tmax)
240  {
241  auto& dyn_npis = this->get_model().parameters.template get<DynamicNPIsInfectedSymptoms>();
242  auto& contact_patterns = this->get_model().parameters.template get<ContactPatterns>();
243  if (dyn_npis.get_thresholds().size() > 0) {
244  auto t = Base::get_result().get_last_time();
245  const auto dt = dyn_npis.get_interval().get();
246 
247  while (t < tmax) {
248  auto dt_eff = std::min({dt, tmax - t, m_t_last_npi_check + dt - t});
249 
250  Base::advance(t + dt_eff);
251  t = t + dt_eff;
252 
253  if (floating_point_greater_equal(t, m_t_last_npi_check + dt)) {
254  auto inf_rel = get_infections_relative(*this, t, this->get_result().get_last_value()) *
255  dyn_npis.get_base_value();
256  auto exceeded_threshold = dyn_npis.get_max_exceeded_threshold(inf_rel);
257  if (exceeded_threshold != dyn_npis.get_thresholds().end() &&
258  (exceeded_threshold->first > m_dynamic_npi.first ||
259  t > double(m_dynamic_npi.second))) { //old npi was weaker or is expired
260  auto t_end = mio::SimulationTime(t + double(dyn_npis.get_duration()));
261  m_dynamic_npi = std::make_pair(exceeded_threshold->first, t_end);
262  mio::implement_dynamic_npis(contact_patterns.get_cont_freq_mat(), exceeded_threshold->second,
263  SimulationTime(t), t_end, [](auto& g) {
264  return mio::make_contact_damping_matrix(g);
265  });
266  }
267 
268  m_t_last_npi_check = t;
269  }
270  }
271 
272  return this->get_result().get_last_value();
273  }
274  else {
275  return Base::advance(tmax);
276  }
277  }
278 
279 private:
280  double m_t_last_npi_check;
281  std::pair<double, SimulationTime> m_dynamic_npi = {-std::numeric_limits<double>::max(), mio::SimulationTime(0)};
282 };

321 template <class Base = mio::Simulation<Model>>
322 auto get_migration_factors(const Simulation<Base>& sim, double /*t*/, const Eigen::Ref<const Eigen::VectorXd>& y)
323 {
324  auto& params = sim.get_model().parameters;
325  //parameters as arrays
326  auto&& t_inc = params.template get<IncubationTime>().array().template cast<double>();
327  auto&& t_ser = params.template get<SerialInterval>().array().template cast<double>();
328  auto&& p_asymp = params.template get<RecoveredPerInfectedNoSymptoms>().array().template cast<double>();
329  auto&& p_inf = params.template get<RiskOfInfectionFromSymptomatic>().array().template cast<double>();
330  auto&& p_inf_max = params.template get<MaxRiskOfInfectionFromSymptomatic>().array().template cast<double>();
331  //slice of InfectedNoSymptoms
332  auto y_car = slice(y, {Eigen::Index(InfectionState::InfectedNoSymptoms),
333  Eigen::Index(size_t(params.get_num_groups())), Eigen::Index(InfectionState::Count)});
334 
335  //compute isolation, same as infection risk from main model
336  auto R3 = 0.5 / (t_inc - t_ser);
337  auto test_and_trace_required = ((1 - p_asymp) * R3 * y_car.array()).sum();
338  auto test_and_trace_capacity = double(params.template get<TestAndTraceCapacity>());
339  auto riskFromInfectedSymptomatic = smoother_cosine(test_and_trace_required, test_and_trace_capacity,
340  test_and_trace_capacity * 5, p_inf.matrix(), p_inf_max.matrix());
341 
342  //set factor for infected
343  auto factors = Eigen::VectorXd::Ones(y.rows()).eval();
344  slice(factors, {Eigen::Index(InfectionState::InfectedSymptoms), Eigen::Index(size_t(params.get_num_groups())),
345  Eigen::Index(InfectionState::Count)})
346  .array() = riskFromInfectedSymptomatic;
347  return factors;
348 }

350 template <class Base = mio::Simulation<Model>>
351 auto test_commuters(Simulation<Base>& sim, Eigen::Ref<Eigen::VectorXd> migrated, double time)
352 {
353  auto& model = sim.get_model();
354  auto nondetection = 1.0;
355  if (time >= model.parameters.get_start_commuter_detection() &&
356  time < model.parameters.get_end_commuter_detection()) {
357  nondetection = (double)model.parameters.get_commuter_nondetection();
358  }
359  for (auto i = AgeGroup(0); i < model.parameters.get_num_groups(); ++i) {
360  auto INSi = model.populations.get_flat_index({i, InfectionState::InfectedNoSymptoms});
361  auto INSCi = model.populations.get_flat_index({i, InfectionState::InfectedNoSymptomsConfirmed});
362  auto ISyi = model.populations.get_flat_index({i, InfectionState::InfectedSymptoms});
363  auto ISyCi = model.populations.get_flat_index({i, InfectionState::InfectedSymptomsConfirmed});
364 
365  //put detected commuters in their own compartment so they don't contribute to infections in their home node
366  sim.get_result().get_last_value()[INSi] -= migrated[INSi] * (1 - nondetection);
367  sim.get_result().get_last_value()[INSCi] += migrated[INSi] * (1 - nondetection);
368  sim.get_result().get_last_value()[ISyi] -= migrated[ISyi] * (1 - nondetection);
369  sim.get_result().get_last_value()[ISyCi] += migrated[ISyi] * (1 - nondetection);
370 
371  //reduce the number of commuters
372  migrated[ISyi] *= nondetection;
373  migrated[INSi] *= nondetection;
374  }
375 }
