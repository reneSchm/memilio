348  //entry to recursively deserialize all parameters in the ParameterSet
349  //IOContext: serializer
350  //IOObject: object that stores the serialized ParameterSet
351  //Rs: IOResult<T> for each Parameter Tag that has already been deserialized
352  template<class IOContext, class IOObject, class... Rs, std::enable_if_t<(sizeof...(Rs) < sizeof...(Tags)), void*> = nullptr>
353  static IOResult<ParameterSet> deserialize_recursive(IOContext& io, IOObject& obj, Rs&&... rs)
354  {
355  //read current parameter, append result to results of previous parameters, recurse to next parameter
356  const size_t I = sizeof...(Rs);
357  using TaggedParameter = std::tuple_element_t<I, decltype(ParameterSet::m_tup)>;
358  auto r = obj.expect_element(TaggedParameter::Tag::name(), mio::Tag<typename TaggedParameter::Type>{});
359  return deserialize_recursive(io, obj, std::forward<Rs>(rs)..., std::move(r));
360  }
