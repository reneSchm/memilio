127 template <class M, std::enable_if_t<is_dynamic_matrix<M>::value, int> = 0>
128 auto slice(M&& m, Seq<Eigen::Index> rows, Seq<Eigen::Index> cols)
129 {
130  assert(rows.start + rows.stride * rows.n <= m.rows());
131  assert(cols.start + cols.stride * cols.n <= m.cols());
132 
133  auto majSpec = std::remove_reference_t<M>::IsRowMajor ? rows : cols;
134  auto minSpec = std::remove_reference_t<M>::IsRowMajor ? cols : rows;
135  auto majStride = majSpec.stride * minor_size(m);
136  auto minStride = minSpec.stride;
137  auto data = m.data() + majSpec.start * minor_size(m) + minSpec.start;
138 
139  return Eigen::Map<CVPlainMatrixT<std::remove_reference_t<M>>, Eigen::Unaligned,
140  Eigen::Stride<Eigen::Dynamic, Eigen::Dynamic>>(data, rows.n, cols.n, {majStride, minStride});
141 }

220 template <class M, bool IsConst = false>
221 class RowMajorIterator
222 {
223 public:
224  using MatrixRef = std::conditional_t<IsConst, const M&, M&>;
225  using MatrixPtr = std::conditional_t<IsConst, const M*, M*>;
226  static_assert(IsConst || details::IsElementReference<M>::value,
227  "Iterator must be const if matrix is not in memory.");
228 
229  using iterator_category = std::random_access_iterator_tag;
230  using value_type = typename M::Scalar;
231  using reference = std::conditional_t<IsConst, const value_type&, value_type&>;
232  using difference_type = Eigen::Index;
233 
234  // Operator-> returns a pointer to an element if the element can be referenced.
235  // If the matrix returns temporaries, a proxy is returned that holds a local copy
236  // and forwards the address of that copy. Adress is pointer-to-const because modifying
237  // the local inside the proxy is almost certainly not what is intended.
238  struct Proxy {
239  value_type value;
240  const value_type* operator->() const
241  {
242  return &value;
243  }
244  };
245  using pointer = std::conditional_t<details::IsElementReference<MatrixRef>::value,
246  std::conditional_t<IsConst, const value_type*, value_type*>, Proxy>;
