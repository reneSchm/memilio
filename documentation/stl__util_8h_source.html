49 template <typename T, typename Pred>
50 typename std::vector<T>::iterator insert_sorted_replace(std::vector<T>& vec, T const& item, Pred pred)
51 {
52  auto bounds = std::equal_range(begin(vec), end(vec), item, pred);
53  auto lb = bounds.first;
54  auto ub = bounds.second;
55  assert(ub - lb <= 1); //input vector contains at most one item that is equal to the new item
56  if (ub - lb == 1) {
57  *lb = item;
58  return lb;
59  }
60  else {
61  return vec.insert(lb, item);
62  }
63 }
