174  template <class Parameter>
175  inline bool matches_parameter() const
176  {
177  if (is_name()) { // match name without "--"
178  return Name<Parameter>::get() == substr(2);
179  }
180  else if (is_alias()) { // match alias without "-"
181  return Alias<Parameter>::get() == substr(1);
182  }
183  else { // not an option
184  return false;
185  }
186  }

202 template <class T, class... Ts>
203 void write_help_impl(const typelist<T, Ts...>, std::ostream& os)
204 {
205  // Max. space and offsets used to print everything nicely.
206  const size_t name_space = 16;
207  const size_t alias_space = 4;
208  const size_t name_indent = 4; // size of " --"
209  const size_t alias_indent = 3; // size of " -"
210  const size_t description_indent = 2;
211  // Get strings. Can assume name is not empty
212  const std::string name = Name<T>::get();
213  const std::string description = Description<T>::get();
214  const std::string alias = Alias<T>::get();
215  // Write name with "--" prefix and " " indent
216  os << " --" << name;
217  if (name.size() <= name_space) {
218  os << std::string(name_space - name.size(), ' ');
219  }
220  else if (description.size() > 0 || alias.size() > 0) {
221  os << "\n" << std::string(name_space + name_indent, ' ');
222  }
223  // Write alias (if available) with "-" prefix and " " indent
224  size_t space = alias_space - alias.size();
225  if (alias.size() > 0) {
226  os << " -" << alias;
227  }
228  else {
229  space += alias_indent;
230  }
231  // Write description (if available) and end line indentation
232  os << std::string(space + description_indent, ' ');
233  os << description << "\n";
234  // Write next entry
235  write_help_impl(typelist<Ts...>(), os);
236 }

265 template <template <class> class Field, class T, class... Ts, class Set>
266 mio::IOResult<void> set_param_impl(const typelist<T, Ts...>, Set& parameters, const char* name, const std::string& args)
267 {
268  if (Field<T>::get() == name) {
269  // read json value from args
270  Json::Value js;
271  std::string errors;
272  Json::CharReaderBuilder builder;
273  const std::unique_ptr<Json::CharReader> parser(builder.newCharReader());
274  parser->parse(args.c_str(), args.c_str() + args.size(), &js, &errors);
275  // deserialize the json value to the parameter's type, then assign it
276  auto result = mio::deserialize_json(js, mio::Tag<typename T::Type>());
277  if (result) {
278  // assign the result to the parameter
279  parameters.template get<T>() = result.value();
280  return mio::success();
281  }
282  else { // deserialize failed
283  // insert more information to the error message
284  std::string msg = "While setting \"" + std::string(name) + "\": " + result.error().message();
285  return mio::failure(result.error().code(), msg);
286  }
287  }
288  else {
289  // try next parameter
290  return set_param_impl<Field>(typelist<Ts...>(), parameters, name, args);
291  }
292 }

448 template <class Set>
449 mio::IOResult<void> command_line_interface(const std::string& executable_name, const int argc, char** argv,
450  Set& parameters, std::ostream& os = std::cout)
451 {
452  using namespace mio::details::cli;
453  // make sure there are no duplicate names or aliases
454  verify_options(parameters);
455  // handle help option, scanning all argumemts before doing anything with them
456  for (int i = 1; i < argc; i++) {
457  if (Identifier(argv[i]).matches_parameter<Help>()) {
458  // print the help dialogue and exit
459  write_help(executable_name, parameters, os);
460  std::exit(0);
461  }
462  }
463  // handle print_option option, scanning all argumemts before doing anything with them
464  for (int i = 1; i < argc; i++) {
465  if (Identifier(argv[i]).matches_parameter<PrintOption>()) {
466  i++; // skip the PrintOption argument
467  for (; i < argc && !Identifier(argv[i]).is_option(); i++) {
468  // try to get the parameter's json value
469  BOOST_OUTCOME_TRY(value, get_param(parameters, argv[i]));
470  // print the name (or alias) and value
471  os << "Option " << argv[i] << ":\n" << value << "\n";
472  }
473  // exit after all values are printed
474  std::exit(0);
475  }
476  }
477  // handle parameter options that require values iteratively. assign given values or return an error
478  int i = 1;
479  while (i < argc) {
480  const Identifier identifier(argv[i]);
481  i++; // go to first argument
482  // assert that the first argument is not an identifier (i.e. name or alias)
483  if (i == argc || Identifier(argv[i]).is_option()) {
484  return mio::failure(mio::StatusCode::OutOfRange, "Missing value for option \"" + identifier + "\".");
485  }
486  // collect all argv's that are not identifiers and set i to the position of the next identifier
487  std::string arguments(argv[i]);
488  i++;
489  for (; (i < argc) && !Identifier(argv[i]).is_option(); i++) {
490  arguments.append(" ").append(argv[i]);
491  }
492  // (try to) set the parameter, to the value given by arguments
493  BOOST_OUTCOME_TRY(set_param(parameters, identifier, arguments));
494  }
495  return mio::success();
496 }
