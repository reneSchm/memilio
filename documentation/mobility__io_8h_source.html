75 template <class Model>
76 IOResult<void> write_graph(const Graph<Model, MigrationParameters>& graph, const std::string& directory,
77  int ioflags = IOF_None)
78 {
79  assert(graph.nodes().size() > 0 && "Graph Nodes are empty");
80 
81  std::string abs_path;
82  BOOST_OUTCOME_TRY(created, create_directory(directory, abs_path));
83 
84  if (created) {
85  log_info("Results are stored in {:s}/results.", mio::get_current_dir_name());
86  }
87  else {
88  log_info("Results are stored in {:s}/results. Files from previous "
89  "graph will be "
90  "overwritten",
91  mio::get_current_dir_name());
92  }
93 
94  //write two files per node
95  //one file that contains outgoing edges from the node
96  //one file for the model (parameters and population)
97  for (auto inode = size_t(0); inode < graph.nodes().size(); ++inode) {
98  //node
99  auto& node = graph.nodes()[inode];
100  BOOST_OUTCOME_TRY(js_node_model, serialize_json(node.property, ioflags));
101  Json::Value js_node(Json::objectValue);
102  js_node["NodeId"] = node.id;
103  js_node["Model"] = js_node_model;
104  auto node_filename = path_join(abs_path, "GraphNode" + std::to_string(inode) + ".json");
105  BOOST_OUTCOME_TRY(write_json(node_filename, js_node));
106 
107  //list of edges
108  auto out_edges = graph.out_edges(inode);
109  if (out_edges.size()) {
110  Json::Value js_edges(Json::arrayValue);
111  for (auto& e : graph.out_edges(inode)) {
112  BOOST_OUTCOME_TRY(js_edge_params, serialize_json(e.property, ioflags));
113  Json::Value js_edge{Json::objectValue};
114  js_edge["StartNodeIndex"] = Json::UInt64(e.start_node_idx);
115  js_edge["EndNodeIndex"] = Json::UInt64(e.end_node_idx);
116  js_edge["Parameters"] = js_edge_params;
117  js_edges.append(std::move(js_edge));
118  }
119  auto edge_filename = path_join(abs_path, "GraphEdges_node" + std::to_string(inode) + ".json");
120  BOOST_OUTCOME_TRY(write_json(edge_filename, js_edges));
121  }
122  }
123 
124  return success();
125 }

135 template <class Model>
136 IOResult<Graph<Model, MigrationParameters>> read_graph(const std::string& directory, int ioflags = IOF_None, bool read_edges = true)
137 {
138  std::string abs_path;
139  if (!file_exists(directory, abs_path)) {
140  log_error("Directory {} does not exist.", directory);
141  return failure(StatusCode::FileNotFound, directory);
142  }
143 
144  auto graph = Graph<Model, MigrationParameters>{};
145 
146  //read nodes, as many as files are available
147  for (auto inode = 0;; ++inode) {
148  auto node_filename = path_join(abs_path, "GraphNode" + std::to_string(inode) + ".json");
149  if (!file_exists(node_filename, node_filename)) {
150  break;
151  }
152  BOOST_OUTCOME_TRY(js_node, read_json(node_filename));
153  if (!js_node["NodeId"].isInt()) {
154  log_error("NodeId field must be an integer.");
155  return failure(StatusCode::InvalidType, node_filename + ", NodeId must be an integer.");
156  }
157  auto node_id = js_node["NodeId"].asInt();
158  BOOST_OUTCOME_TRY(model, deserialize_json(js_node["Model"], Tag<Model>{}, ioflags));
159  graph.add_node(node_id, model);
160  }
161 
162  //read edges; nodes must already be available for that)
163  if(read_edges){
164  for (auto inode = size_t(0); inode < graph.nodes().size(); ++inode) {
165  //list of edges
166  auto edge_filename = path_join(abs_path, "GraphEdges_node" + std::to_string(inode) + ".json");
167  BOOST_OUTCOME_TRY(js_edges, read_json(edge_filename));
168 
169  for (auto& e : js_edges) {
170  auto start_node_idx = inode;
171  auto js_end_node_idx = e["EndNodeIndex"];
172  if (!js_end_node_idx.isUInt64()) {
173  log_error("EndNodeIndex must be an integer.");
174  return failure(StatusCode::InvalidType, edge_filename + ", EndNodeIndex must be an integer.");
175  }
176  auto end_node_idx = js_end_node_idx.asUInt64();
177  if (end_node_idx >= graph.nodes().size()) {
178  log_error("EndNodeIndex not in range of number of graph nodes.");
179  return failure(StatusCode::OutOfRange,
180  edge_filename + ", EndNodeIndex not in range of number of graph nodes.");
181  }
182  BOOST_OUTCOME_TRY(parameters, deserialize_json(e["Parameters"], Tag<MigrationParameters>{}, ioflags));
183  graph.add_edge(start_node_idx, end_node_idx, parameters);
184  }
185  }
186  }
187 
188  return success(graph);
189 }
