396 IOResult<void> set_divi_data(std::vector<Model>& model, const std::string& path, const std::vector<int>& vregion,
397  Date date, double scaling_factor_icu)
398 {
399  std::vector<double> sum_mu_I_U(vregion.size(), 0);
400  std::vector<std::vector<double>> mu_I_U{model.size()};
401  for (size_t region = 0; region < vregion.size(); region++) {
402  auto num_groups = model[region].parameters.get_num_groups();
403  for (auto i = AgeGroup(0); i < num_groups; i++) {
404  sum_mu_I_U[region] += model[region].parameters.template get<CriticalPerSevere>()[i] *
405  model[region].parameters.template get<SeverePerInfectedSymptoms>()[i];
406  mu_I_U[region].push_back(model[region].parameters.template get<CriticalPerSevere>()[i] *
407  model[region].parameters.template get<SeverePerInfectedSymptoms>()[i]);
408  }
409  }
410  std::vector<double> num_icu(model.size(), 0.0);
411  BOOST_OUTCOME_TRY(read_divi_data(path, vregion, date, num_icu));
412 
413  for (size_t region = 0; region < vregion.size(); region++) {
414  auto num_groups = model[region].parameters.get_num_groups();
415  for (auto i = AgeGroup(0); i < num_groups; i++) {
416  model[region].populations[{i, InfectionState::InfectedCriticalNaive}] =
417  scaling_factor_icu * num_icu[region] * mu_I_U[region][(size_t)i] / sum_mu_I_U[region];
418  }
419  }
420 
421  return success();
422 }

444  std::vector<std::vector<double>> num_rec(model.size(), std::vector<double>(num_age_groups, 0.0));
445

450  auto num_groups = model[region].parameters.get_num_groups();
451  for (auto i = AgeGroup(0); i < num_groups; i++) {

453  double S_v =
454  std::min(model[region].parameters.template get<DailyFullVaccination>()[{i, SimulationDay(0)}] +
455  num_rec[region][size_t(i)],
456  num_population[region][size_t(i)]);
457  double S_pv =
458  std::max(model[region].parameters.template get<DailyFirstVaccination>()[{i, SimulationDay(0)}] -
459  model[region].parameters.template get<DailyFullVaccination>()[{i, SimulationDay(0)}],
460  0.0); // use std::max with 0
461  double S;
462  if (num_population[region][size_t(i)] - S_pv - S_v < 0.0) {
463  log_warning("Number of vaccinated persons greater than population in county {}, age group {}.",
464  region, size_t(i));
465  S = 0.0;
466  S_v = num_population[region][size_t(i)] - S_pv;
467  }
468  else {
469  S = num_population[region][size_t(i)] - S_pv - S_v;
470  }
471 
472  double denom_E =
473  1 / (S + S_pv * model[region].parameters.template get<ReducExposedPartialImmunity>()[i] +
474  S_v * model[region].parameters.template get<ReducExposedImprovedImmunity>()[i]);
475  double denom_C =
476  1 / (S + S_pv * model[region].parameters.template get<ReducExposedPartialImmunity>()[i] +
477  S_v * model[region].parameters.template get<ReducExposedImprovedImmunity>()[i]);
478  double denom_I =
479  1 / (S + S_pv * model[region].parameters.template get<ReducInfectedSymptomsPartialImmunity>()[i] +
480  S_v * model[region].parameters.template get<ReducInfectedSymptomsImprovedImmunity>()[i]);
481  double denom_HU =
482  1 /
483  (S +
484  S_pv * model[region].parameters.template get<ReducInfectedSevereCriticalDeadPartialImmunity>()[i] +
485  S_v * model[region].parameters.template get<ReducInfectedSevereCriticalDeadImprovedImmunity>()[i]);

488  S * model[region].populations[{i, InfectionState::ExposedNaive}] * denom_E;
489  model[region].populations[{i, InfectionState::ExposedPartialImmunity}] =
490  S_pv * model[region].parameters.template get<ReducExposedPartialImmunity>()[i] *
491  model[region].populations[{i, InfectionState::ExposedPartialImmunity}] * denom_E;
492  model[region].populations[{i, InfectionState::ExposedImprovedImmunity}] =
493  S_v * model[region].parameters.template get<ReducExposedImprovedImmunity>()[i] *
494  model[region].populations[{i, InfectionState::ExposedImprovedImmunity}] * denom_E;
495 
496  model[region].populations[{i, InfectionState::InfectedNoSymptomsNaive}] =
497  S * model[region].populations[{i, InfectionState::InfectedNoSymptomsNaive}] * denom_C;
498  model[region].populations[{i, InfectionState::InfectedNoSymptomsPartialImmunity}] =
499  S_pv * model[region].populations[{i, InfectionState::InfectedNoSymptomsPartialImmunity}] * denom_C;
500  model[region].populations[{i, InfectionState::InfectedNoSymptomsImprovedImmunity}] =
501  S_v * model[region].populations[{i, InfectionState::InfectedNoSymptomsImprovedImmunity}] * denom_C;
502 
503  model[region].populations[{i, InfectionState::InfectedNoSymptomsNaiveConfirmed}] =
504  S * model[region].populations[{i, InfectionState::InfectedNoSymptomsNaiveConfirmed}] * denom_C;
505  model[region].populations[{i, InfectionState::InfectedNoSymptomsPartialImmunityConfirmed}] =
506  S_pv * model[region].populations[{i, InfectionState::InfectedNoSymptomsPartialImmunityConfirmed}] *

511

544  model[region].populations[{i, InfectionState::InfectedCriticalNaive}] * denom_HU;

546  S * model[region].populations[{i, InfectionState::InfectedCriticalNaive}] * denom_HU;

555  model[region].populations[{i, InfectionState::InfectedSymptomsPartialImmunityConfirmed}] +
556  model[region].populations[{i, InfectionState::InfectedSymptomsImprovedImmunityConfirmed}] +

558  model[region].populations[{i, InfectionState::InfectedSeverePartialImmunity}] +
559  model[region].populations[{i, InfectionState::InfectedSevereImprovedImmunity}] +
560  model[region].populations[{i, InfectionState::InfectedCriticalNaive}] +
561  model[region].populations[{i, InfectionState::InfectedCriticalPartialImmunity}] +
562  model[region].populations[{i, InfectionState::InfectedCriticalImprovedImmunity}] +
563  model[region].populations[{i, InfectionState::DeadNaive}] +
564  model[region].populations[{i, InfectionState::DeadPartialImmunity}] +
565  model[region].populations[{i, InfectionState::DeadImprovedImmunity}]);
566

574  model[region].populations[{i, InfectionState::InfectedNoSymptomsPartialImmunity}] -
575  model[region].populations[{i, InfectionState::InfectedNoSymptomsPartialImmunityConfirmed}] -
576  model[region].populations[{i, InfectionState::InfectedSymptomsPartialImmunity}] -
577  model[region].populations[{i, InfectionState::InfectedSymptomsPartialImmunityConfirmed}] -
578  model[region].populations[{i, InfectionState::InfectedSeverePartialImmunity}] -
579  model[region].populations[{i, InfectionState::InfectedCriticalPartialImmunity}]);
580 
581  model[region].populations.template set_difference_from_group_total<AgeGroup>(
582  {i, InfectionState::SusceptibleNaive}, num_population[region][size_t(i)]);
583  }
584 
585  for (auto i = AgeGroup(0); i < AgeGroup(6); i++) {
586  for (auto j = Index<InfectionState>(0); j < InfectionState::Count; ++j) {
587  if (model[region].populations[{i, j}] < 0) {
588  log_warning("Compartment at age group {}, infection state {}, is negative: {}", size_t(i),
589  size_t(j), model[region].populations[{i, j}] / num_population[region][size_t(i)]);
590  }
591  }
592  }
593  }
594  else {
595  log_warning("No population data available for region " + std::to_string(region) +
596  ". Population data has not been set.");
597  }
598  }
599 
600  return success();
601 }
