145 template <class NodePropertyT, class EdgePropertyT>
146 class Graph
147  //ensure correct std::is_copy_constructible; it's not correct by default because the nodes and edges are stored in std::vector
148  : not_copyable_if_t<
149  !conjunction<std::is_copy_constructible<NodePropertyT>, std::is_copy_constructible<EdgePropertyT>>::value>
150 {
151 public:
152  using NodeProperty = NodePropertyT;
153  using EdgeProperty = EdgePropertyT;
154 
158  template <class... Args>
159  Node<NodePropertyT>& add_node(int id, Args&&... args)
160  {
161  m_nodes.emplace_back(id, std::forward<Args>(args)...);
162  return m_nodes.back();
163  }
164 
168  template <class... Args>
169  Edge<EdgePropertyT>& add_edge(size_t start_node_idx, size_t end_node_idx, Args&&... args)
170  {
171  assert(m_nodes.size() > start_node_idx && m_nodes.size() > end_node_idx);
172  return *insert_sorted_replace(m_edges,
173  Edge<EdgePropertyT>(start_node_idx, end_node_idx, std::forward<Args>(args)...),
174  [](auto&& e1, auto&& e2) {
175  return e1.start_node_idx == e2.start_node_idx
176  ? e1.end_node_idx < e2.end_node_idx
177  : e1.start_node_idx < e2.start_node_idx;
178  });
179  }
180

186  return make_range(begin(m_nodes), end(m_nodes));
187  }

262 template <class TestAndTrace, class ContactPattern, class Model, class MigrationParams, class Parameters,
263  class ReadFunction, class NodeIdFunction>
264 IOResult<void>
265 set_nodes(const Parameters& params, Date start_date, Date end_date, const fs::path& data_dir,
266  const std::string& population_data_path, bool is_node_for_county, Graph<Model, MigrationParams>& params_graph,
267  ReadFunction&& read_func, NodeIdFunction&& node_func, const std::vector<double>& scaling_factor_inf,
268  double scaling_factor_icu, double tnt_capacity_factor, int num_days = 0, bool export_time_series = false)
269 {
270  BOOST_OUTCOME_TRY(node_ids, node_func(population_data_path, is_node_for_county));
271  std::vector<Model> nodes(node_ids.size(), Model(int(size_t(params.get_num_groups()))));
272  for (auto& node : nodes) {
273  node.parameters = params;
274  }
275 
276  BOOST_OUTCOME_TRY(read_func(nodes, start_date, node_ids, scaling_factor_inf, scaling_factor_icu, data_dir.string(),
277  num_days, export_time_series));
278 
279  for (size_t node_idx = 0; node_idx < nodes.size(); ++node_idx) {
280 
281  auto tnt_capacity = nodes[node_idx].populations.get_total() * tnt_capacity_factor;
282 
283  //local parameters
284  auto& tnt_value = nodes[node_idx].parameters.template get<TestAndTrace>();
285  tnt_value = UncertainValue(0.5 * (1.2 * tnt_capacity + 0.8 * tnt_capacity));
286  tnt_value.set_distribution(mio::ParameterDistributionUniform(0.8 * tnt_capacity, 1.2 * tnt_capacity));
287 
288  auto id = 0;
289  if (is_node_for_county) {
290  id = int(regions::CountyId(node_ids[node_idx]));
291  }
292  else {
293  id = int(regions::DistrictId(node_ids[node_idx]));
294  }
295  //holiday periods
296  auto holiday_periods = regions::get_holidays(regions::get_state_id(id), start_date, end_date);
297  auto& contacts = nodes[node_idx].parameters.template get<ContactPattern>();
298  contacts.get_school_holidays() =
299  std::vector<std::pair<mio::SimulationTime, mio::SimulationTime>>(holiday_periods.size());
300  std::transform(
301  holiday_periods.begin(), holiday_periods.end(), contacts.get_school_holidays().begin(), [=](auto& period) {
302  return std::make_pair(mio::SimulationTime(mio::get_offset_in_days(period.first, start_date)),
303  mio::SimulationTime(mio::get_offset_in_days(period.second, start_date)));
304  });
305 
306  //uncertainty in populations
307  for (auto i = mio::AgeGroup(0); i < params.get_num_groups(); i++) {
308  for (auto j = Index<typename Model::Compartments>(0); j < Model::Compartments::Count; ++j) {
309  auto& compartment_value = nodes[node_idx].populations[{i, j}];
310  compartment_value =
311  UncertainValue(0.5 * (1.1 * double(compartment_value) + 0.9 * double(compartment_value)));
312  compartment_value.set_distribution(mio::ParameterDistributionUniform(0.9 * double(compartment_value),
313  1.1 * double(compartment_value)));
314  }
315  }
316 
317  params_graph.add_node(node_ids[node_idx], nodes[node_idx]);
318  }
319  return success();
320 }

332 template <class ContactLocation, class Model, class MigrationParams, class MigrationCoefficientGroup,
333  class InfectionState, class ReadFunction>
334 IOResult<void> set_edges(const fs::path& data_dir, Graph<Model, MigrationParams>& params_graph,
335  std::initializer_list<InfectionState>& migrating_compartments, size_t contact_locations_size,
336  ReadFunction&& read_func,
337  std::vector<ScalarType> commuting_weights = std::vector<ScalarType>{})
338 {
339  // mobility between nodes
340  BOOST_OUTCOME_TRY(mobility_data_commuter,
341  read_func((data_dir / "mobility" / "commuter_migration_scaled.txt").string()));
342  BOOST_OUTCOME_TRY(mobility_data_twitter, read_func((data_dir / "mobility" / "twitter_scaled_1252.txt").string()));
343  if (mobility_data_commuter.rows() != Eigen::Index(params_graph.nodes().size()) ||
344  mobility_data_commuter.cols() != Eigen::Index(params_graph.nodes().size()) ||
345  mobility_data_twitter.rows() != Eigen::Index(params_graph.nodes().size()) ||
346  mobility_data_twitter.cols() != Eigen::Index(params_graph.nodes().size())) {
347  return mio::failure(mio::StatusCode::InvalidValue,
348  "Mobility matrices do not have the correct size. You may need to run "
349  "transformMobilitydata.py from pycode memilio epidata package.");
350  }
351 
352  for (size_t county_idx_i = 0; county_idx_i < params_graph.nodes().size(); ++county_idx_i) {
353  for (size_t county_idx_j = 0; county_idx_j < params_graph.nodes().size(); ++county_idx_j) {
354  auto& populations = params_graph.nodes()[county_idx_i].property.populations;
355  // mobility coefficients have the same number of components as the contact matrices.
356  // so that the same NPIs/dampings can be used for both (e.g. more home office => fewer commuters)
357  auto mobility_coeffs = MigrationCoefficientGroup(contact_locations_size, populations.numel());
358  auto num_age_groups = (size_t)params_graph.nodes()[county_idx_i].property.parameters.get_num_groups();
359  commuting_weights =
360  (commuting_weights.size() == 0 ? std::vector<ScalarType>(num_age_groups, 1.0) : commuting_weights);
361  //commuters
362  auto working_population = 0.0;
363  for (auto age = AgeGroup(0); age < populations.template size<mio::AgeGroup>(); ++age) {
364  working_population += populations.get_group_total(age) * commuting_weights[size_t(age)];
365  }
366  auto commuter_coeff_ij = mobility_data_commuter(county_idx_i, county_idx_j) /
367  working_population; //data is absolute numbers, we need relative
368  for (auto age = AgeGroup(0); age < populations.template size<mio::AgeGroup>(); ++age) {
369  for (auto compartment : migrating_compartments) {
370  auto coeff_index = populations.get_flat_index({age, compartment});
371  mobility_coeffs[size_t(ContactLocation::Work)].get_baseline()[coeff_index] =
372  commuter_coeff_ij * commuting_weights[size_t(age)];
373  }
374  }
375  //others
376  auto total_population = populations.get_total();
377  auto twitter_coeff = mobility_data_twitter(county_idx_i, county_idx_j) /
378  total_population; //data is absolute numbers, we need relative
379  for (auto age = AgeGroup(0); age < populations.template size<mio::AgeGroup>(); ++age) {
380  for (auto compartment : migrating_compartments) {
381  auto coeff_idx = populations.get_flat_index({age, compartment});
382  mobility_coeffs[size_t(ContactLocation::Other)].get_baseline()[coeff_idx] = twitter_coeff;
383  }
384  }
385 
386  //only add edges with mobility above thresholds for performance
387  //thresholds are chosen empirically so that more than 99% of mobility is covered, approx. 1/3 of the edges
388  if (commuter_coeff_ij > 4e-5 || twitter_coeff > 1e-5) {
389  params_graph.add_edge(county_idx_i, county_idx_j, std::move(mobility_coeffs));
390  }
391  }
392  }
393 
394  return success();
395 }
