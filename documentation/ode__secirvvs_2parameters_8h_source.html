638  {
639  int corrected = false;
640  if (this->get<Seasonality>() < 0.0 || this->get<Seasonality>() > 0.5) {
641  log_warning("Constraint check: Parameter Seasonality changed from {:0.4f} to {:d}",
642  this->get<Seasonality>(), 0);

644  corrected = true;
645  }

647  if (this->get<ICUCapacity>() < 0.0) {
648  log_warning("Constraint check: Parameter ICUCapacity changed from {:0.4f} to {:d}",

654  double tol_times = 1e-4; // accepted tolerance for compartment stays
655 
656  for (auto i = AgeGroup(0); i < AgeGroup(m_num_groups); ++i) {
657 
658  if (this->get<IncubationTime>()[i] < 2 * tol_times) {
659  log_warning("Constraint check: Parameter IncubationTime changed from {:.4f} to {:.4f}",
660  this->get<IncubationTime>()[i], tol_times);
661  this->get<IncubationTime>()[i] = 2 * tol_times;
662  corrected = true;
663  }
664 
665  if (2 * this->get<SerialInterval>()[i] < this->get<IncubationTime>()[i] + tol_times) {
666  log_warning("Constraint check: Parameter SerialInterval changed from {:.4f} to {:.4f}",
667  this->get<SerialInterval>()[i], 0.5 * this->get<IncubationTime>()[i] + tol_times / 2);
668  this->get<SerialInterval>()[i] = 0.5 * this->get<IncubationTime>()[i] + tol_times / 2;
669  corrected = true;
