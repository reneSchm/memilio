141  auto num_remaining = m_block_size * block_idx - m_num_generated;

388  template <class RNG>
389  result_type operator()(RNG& rng, param_type p)
390  {
391  auto weights = p.weights();
392  if (weights.size() <= 1) {
393  return 0;
394  }
395  auto sum = std::accumulate(weights.begin(), weights.end(), 0.0);
396  auto u =
397  std::uniform_real_distribution<double>()(rng, std::uniform_real_distribution<double>::param_type{0.0, sum});
398  auto intermediate_sum = 0.0;
399  for (size_t i = 0; i < weights.size(); ++i) {
400  intermediate_sum += weights.get_ptr()[i];
401  if (u < intermediate_sum) {
402  return i;
403  }
404  }
405  assert(false && "this should never happen.");
406  return result_type(-1);
407  }
408 
409 private:
410  param_type m_params;
