216  /*
217  * @brief verification that at least 99% of the density
218  * function lie in the interval defined by the boundaries
219  */
220  bool check_quantiles(double& mean, double& standard_dev)
221  {
222  bool changed = false;
223 
224  changed = adapt_mean(mean);
225 
226  changed = adapt_standard_dev(standard_dev);
227 
228  if (changed && m_log_stddev_change) {
229  log_warning("Standard deviation reduced to fit 99% of the distribution within [lowerbound,upperbound].");
230  }
231 
232  return changed;
233  }
234 
235  bool adapt_mean(double& mean)
236  {
237  bool changed = false;
238  if (mean < m_lower_bound || mean > m_upper_bound) {
239  mean = 0.5 * (m_upper_bound - m_lower_bound);
240  changed = true;
241  }
242  return changed;
243  }
244 
245  // ensure that 0.99 % of the distribution are within lower bound and upper bound
246  bool adapt_standard_dev(double& standard_dev)
247  {
248  bool changed = false;
249  if (m_mean + standard_dev * m_quantile > m_upper_bound) {
250  standard_dev = (m_upper_bound - m_mean) / m_quantile;
251  changed = true;
252  }
253  if (m_mean - standard_dev * m_quantile < m_lower_bound) {
254  standard_dev = (m_mean - m_lower_bound) / m_quantile;
255  changed = true;
256  }
257 
258  return changed;
259  }
260

281  /*
282  * @brief gets a sample of a normally distributed variable
283  * before sampling, it is verified that at least 99% of the
284  * density function lie in the interval defined by the boundaries
285  * otherwise the normal distribution is adapted
286  */
287  double get_rand_sample() override
288  {
289  //If ub = lb, sampling can only be succesful if mean = lb and dev = 0.
290  //But this degenerate normal distribution is not allowed by the c++ standard.
291  if (m_upper_bound == m_lower_bound) {
292  return m_lower_bound;
293  }
294 
295  if (check_quantiles(m_mean, m_standard_dev) || m_distribution.mean() != m_mean ||
296  m_distribution.stddev() != m_standard_dev) {
297  m_distribution = std::normal_distribution<double>{m_mean, m_standard_dev};
298  }
299 
300  int i = 0;
301  int retries = 10;
302  double rnumb = m_distribution(thread_local_rng());
303  while ((rnumb > m_upper_bound || rnumb < m_lower_bound) && i < retries) {
304  rnumb = m_distribution(thread_local_rng());
305  i++;
306  if (i == retries) {
307  log_warning("Not successfully sampled within [min,max].");
308  if (rnumb > m_upper_bound) {
309  rnumb = m_upper_bound;
310  }
311  else {
312  rnumb = m_lower_bound;
313  }
314  }
315  }
316  return rnumb;
317  }

483 template <class IOContext>
484 IOResult<std::shared_ptr<ParameterDistribution>> deserialize_internal(IOContext& io,
485  Tag<std::shared_ptr<ParameterDistribution>>)
486 {
487  auto obj = io.expect_object("ParameterDistribution");
488  auto type = obj.expect_element("Type", Tag<std::string>{});
489  if (type) {
490  if (type.value() == "Uniform") {
491  BOOST_OUTCOME_TRY(r, ParameterDistributionUniform::deserialize_elements(io, obj));
492  return std::make_shared<ParameterDistributionUniform>(r);
493  }
494  else if (type.value() == "Normal") {
495  BOOST_OUTCOME_TRY(r, ParameterDistributionNormal::deserialize_elements(io, obj));
496  return std::make_shared<ParameterDistributionNormal>(r);
497  }
498  else {
499  return failure(StatusCode::InvalidValue, "Type of ParameterDistribution " + type.value() + " not valid.");
500  }
501  }
502  return failure(type.error());
503 }
