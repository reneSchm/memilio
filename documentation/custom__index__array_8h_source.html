79 template <size_t I, typename Index>

87 template <size_t I, typename Index>
88 std::enable_if_t<(I < (Index::size - 1)), std::pair<size_t, size_t>> flatten_index(Index const& indices,
89  Index const& dimensions)
90 {
91  assert(mio::get<I>(indices) < mio::get<I>(dimensions));
92 
93  size_t val, prod;
94  std::tie(val, prod) = flatten_index<I + 1>(indices, dimensions);
95 
96  return {val + (size_t)mio::get<I>(indices) * prod, prod * (size_t)mio::get<I>(dimensions)};
97 }

600  inline Slice::difference_type outer_offset(difference_type const& inner) const
601  {
602 
603  // calculate the outer offset from the inner offset
604 
605  // first unravel the inner index into an index (i,j,k) for a 3-dim array with dims (dl, idx_sequence.n, dr)
606  auto dv = std::div(inner, seq.n * dr);
607  difference_type i = dv.quot;
608  dv = std::div(dv.rem, dr);
609  difference_type j = dv.quot * seq.stride + seq.start;
610  difference_type k = dv.rem;
611 
612  // then flatten the index for a 3-dim array with dims (dl, di, dr)
613  return i * di * dr + j * dr + k;
614  }
