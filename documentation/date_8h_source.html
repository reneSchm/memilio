175 inline Date offset_date_by_days(Date date, int offset_days)
176 {
177  auto year = date.year;
178  auto month = date.month;
179  auto day = date.day;
180  assert(month > 0 && month < 13 && day > 0 && day < 32);
181 
182  std::array<int, 12> month_len;
183  if (((year % 4 == 0) && (year % 100 != 0)) || (year % 400 == 0)) {
184  // leap year
185  month_len = {31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
186  }
187  else {
188  month_len = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
189  }
190  if (day + offset_days > 0 && day + offset_days <= month_len[month - 1]) {
191  return {year, month, day + offset_days};
192  }
193  else {
194  std::array<int, 12> part_sum;
195  std::partial_sum(month_len.begin(), month_len.end(), part_sum.begin());
196 
197  int day_in_year = day + offset_days;
198  if (month > 1) {
199  // take month-2 since end of last month has to be found and due to start at 0 of C++ (against January=1 in date format)
200  day_in_year += part_sum[month - 2];
201  }
202 
203  if (day_in_year > 0 && day_in_year <= part_sum[11]) {
204  auto iter = std::find_if(part_sum.begin(), part_sum.end(), [day_in_year](auto s) {
205  return day_in_year <= s;
206  });
207  int i = static_cast<int>(iter - part_sum.begin());
208  return {year, i + 1, day_in_year - (i > 0 ? part_sum[static_cast<size_t>(i - 1)] : 0)};
209  }
210  else {
211  if (day_in_year > 0) {
212  return offset_date_by_days({year + 1, 1, 1}, day_in_year - part_sum[11] - 1);
213  }
214  else {
215  return offset_date_by_days({year - 1, 12, 31}, day_in_year);
216  }
217  }
218  }
219 }

262 inline int get_offset_in_days(Date date1, Date date2)
263 {
264  auto year1 = date1.year;
265  auto month1 = date1.month;
266  auto day1 = date1.day;
267 
268  auto year2 = date2.year;
269  auto month2 = date2.month;
270  auto day2 = date2.day;
271 
272  if (year1 == year2 && month1 == month2) {
273  return day1 - day2;
274  }
275  else {
276  int day_in_year1 = get_day_in_year(date1);
277  int day_in_year2 = get_day_in_year(date2);
278 
279  if (year1 < year2) {
280  int sum_days = 0;
281  for (int i = year1; i < year2; i++) {
282  sum_days += 365;
283  if (((i % 4 == 0) && (i % 100 != 0)) || (i % 400 == 0)) {
284  sum_days += 1;
285  }
286  }
287  return -(sum_days - day_in_year1) - day_in_year2;
288  }
289  else if (year1 > year2) {
290  int sum_days = 0;
291  for (int i = year2; i < year1; i++) {
292  sum_days += 365;
293  if (((i % 4 == 0) && (i % 100 != 0)) || (i % 400 == 0)) {
294  sum_days += 1;
295  }
296  }
297  return day_in_year1 + sum_days - day_in_year2;
298  }
299  else {
300 
301  return day_in_year1 - day_in_year2;
302  }
303  }
