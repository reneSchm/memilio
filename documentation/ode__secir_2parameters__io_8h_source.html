178  t_InfectedSevere[county].push_back(static_cast<int>(
179  std::round(model[county].parameters.template get<TimeInfectedSevere>()[AgeGroup(group)])));
180  t_InfectedCritical[county].push_back(static_cast<int>(
181  std::round(model[county].parameters.template get<TimeInfectedCritical>()[(AgeGroup)group])));
182  t_InfectedNoSymptoms[county].push_back(static_cast<int>(
183  std::round(2 * (model[county].parameters.template get<IncubationTime>()[AgeGroup(group)] -
184  model[county].parameters.template get<SerialInterval>()[AgeGroup(group)]))));
185  t_Exposed[county].push_back(static_cast<int>(
186  std::round(2 * model[county].parameters.template get<SerialInterval>()[AgeGroup(group)] -
187  model[county].parameters.template get<IncubationTime>()[AgeGroup(group)])));
188  t_InfectedSymptoms[county].push_back(static_cast<int>(
189  std::round(model[county].parameters.template get<TimeInfectedSymptoms>()[AgeGroup(group)])));
190  t_InfectedSevere[county].push_back(static_cast<int>(
191  std::round(model[county].parameters.template get<TimeInfectedSevere>()[AgeGroup(group)])));

252  rki_data[i][j]((size_t)InfectionState::InfectedSymptomsConfirmed +

254  rki_data[i][j]((size_t)InfectionState::InfectedSevere + (size_t)InfectionState::Count * age) =
