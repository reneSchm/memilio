415 template <class Sim>
416 void MigrationEdge::apply_migration(double t, double dt, SimulationNode<Sim>& node_from, SimulationNode<Sim>& node_to)
417 {
418  //check dynamic npis
419  if (m_t_last_dynamic_npi_check == -std::numeric_limits<double>::infinity()) {
420  m_t_last_dynamic_npi_check = node_from.get_t0();
421  }
422 
423  auto& dyn_npis = m_parameters.get_dynamic_npis_infected();
424  if (dyn_npis.get_thresholds().size() > 0 &&
425  floating_point_greater_equal(t, m_t_last_dynamic_npi_check + dyn_npis.get_interval().get())) {
426  auto inf_rel = get_infections_relative(node_from, t, node_from.get_last_state()) * dyn_npis.get_base_value();
427  auto exceeded_threshold = dyn_npis.get_max_exceeded_threshold(inf_rel);
428  if (exceeded_threshold != dyn_npis.get_thresholds().end() &&
429  (exceeded_threshold->first > m_dynamic_npi.first ||
430  t > double(m_dynamic_npi.second))) { //old NPI was weaker or is expired
431  auto t_end = SimulationTime(t + double(dyn_npis.get_duration()));
432  m_dynamic_npi = std::make_pair(exceeded_threshold->first, t_end);
433  implement_dynamic_npis(
434  m_parameters.get_coefficients(), exceeded_threshold->second, SimulationTime(t), t_end, [this](auto& g) {
435  return make_migration_damping_vector(m_parameters.get_coefficients().get_shape(), g);
436  });
437  }
438  m_t_last_dynamic_npi_check = t;
439  }
440 
441  //returns
442  for (Eigen::Index i = m_return_times.get_num_time_points() - 1; i >= 0; --i) {
443  if (m_return_times.get_time(i) <= t) {
444  auto v0 = find_value_reverse(node_to.get_result(), m_migrated.get_time(i), 1e-10, 1e-10);
445  assert(v0 != node_to.get_result().rend() && "unexpected error.");
446  calculate_migration_returns(m_migrated[i], node_to.get_simulation(), *v0, m_migrated.get_time(i), dt);
447 
448  //the lower-order return calculation may in rare cases produce negative compartments,
449  //especially at the beginning of the simulation.
450  //fix by subtracting the supernumerous returns from the biggest compartment of the age group.
451  Eigen::VectorXd remaining_after_return = (node_to.get_result().get_last_value() - m_migrated[i]).eval();
452  for (Eigen::Index j = 0; j < node_to.get_result().get_last_value().size(); ++j) {
453  if (remaining_after_return(j) < 0) {
454  auto num_comparts = (Eigen::Index)Sim::Model::Compartments::Count;
455  auto group = Eigen::Index(j / num_comparts);
456  auto compart = j % num_comparts;
457  log(remaining_after_return(j) < -1e-3 ? LogLevel::warn : LogLevel::info,
458  "Underflow during migration returns at time {}, compartment {}, age group {}: {}", t, compart,
459  group, remaining_after_return(j));
460  Eigen::Index max_index;
461  slice(remaining_after_return, {group * num_comparts, num_comparts}).maxCoeff(&max_index);
462  log_info("Transferring to compartment {}", max_index);
463  max_index += group * num_comparts;
464  m_migrated[i](max_index) -= remaining_after_return(j);
465  m_migrated[i](j) += remaining_after_return(j);
466  }
467  }
468  node_from.get_result().get_last_value() += m_migrated[i];
469  node_to.get_result().get_last_value() -= m_migrated[i];
470  m_migrated.remove_time_point(i);
471  m_return_times.remove_time_point(i);
472  }
473  }
474 
475  if (!m_return_migrated && (m_parameters.get_coefficients().get_matrix_at(t).array() > 0.0).any()) {
476  //normal daily migration
477  m_migrated.add_time_point(
478  t, (node_from.get_last_state().array() * m_parameters.get_coefficients().get_matrix_at(t).array() *
479  get_migration_factors(node_from, t, node_from.get_last_state()).array())
480  .matrix());
481  m_return_times.add_time_point(t + dt);
482 
483  test_commuters(node_from, m_migrated.get_last_value(), t);
484 
485  node_to.get_result().get_last_value() += m_migrated.get_last_value();
486  node_from.get_result().get_last_value() -= m_migrated.get_last_value();
487  }
488  m_return_migrated = !m_return_migrated;
489 }
