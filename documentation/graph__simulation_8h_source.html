61  void advance(double t_max = 1.0)
62  {
63  auto dt = m_dt;
64  while (m_t < t_max) {
65  if (m_t + dt > t_max) {
66  dt = t_max - m_t;
67  }
68 
69  for (auto& n : m_graph.nodes()) {
70  m_node_func(m_t, dt, n.property);
71  }
72 
73  m_t += dt;
74 
75  for (auto& e : m_graph.edges()) {
76  m_edge_func(m_t, dt, e.property, m_graph.nodes()[e.start_node_idx].property,
77  m_graph.nodes()[e.end_node_idx].property);
78  }
79  }
80  }

117 class GraphSimulationStochastic
118  : public GraphSimulationBase<Graph,
119  std::function<void(typename Graph::EdgeProperty&, size_t,
120  typename Graph::NodeProperty&, typename Graph::NodeProperty&)>>
121 {
122  using Base =
123  GraphSimulationBase<Graph, std::function<void(typename Graph::EdgeProperty&, size_t,
124  typename Graph::NodeProperty&, typename Graph::NodeProperty&)>>;
125  using node_function = typename Base::node_function;

145  void advance(double t_max)
146  {
147  //draw normalized waiting time
148  ScalarType normalized_waiting_time = ExponentialDistribution<ScalarType>::get_instance()(1.0);
149  std::vector<ScalarType> dt_cand(Base::m_graph.nodes().size());
150  ScalarType cumulative_rate = 0; //cumulative transition rate
151  size_t parameters_per_edge =
152  size_t(Base::m_graph.edges()[0].property.get_parameters().get_coefficients().get_shape().rows());
153  std::vector<ScalarType> transition_rates(parameters_per_edge * Base::m_graph.edges().size());
154  while (Base::m_t < t_max) {
155  Base::m_dt = std::min({Base::m_dt, t_max - Base::m_t});
156  //calculate current transition rates and cumulative rate
157  cumulative_rate = get_cumulative_transition_rate();
158  if (cumulative_rate * Base::m_dt >
159  normalized_waiting_time) { //at least one transition event during current time step
160  do {
161  //evaluate rates
162  get_rates(m_rates);
163  //draw transition event
164  size_t event = mio::DiscreteDistribution<size_t>::get_instance()(m_rates);
165  //edge that performs transition event
166  auto& event_edge = Base::m_graph.edges()[event / parameters_per_edge];
167  //index for compartment and age group migrating
168  auto flat_index = event % parameters_per_edge;
169 
170  //advance nodes until t + (waiting_time / cumulative_rate)
171  for (size_t node_iter = 0; node_iter < Base::m_graph.nodes().size(); ++node_iter) {
172  auto& node = Base::m_graph.nodes()[node_iter];
173  Base::m_node_func(Base::m_t, normalized_waiting_time / cumulative_rate, node.property);
174  }
175 
176  //advance time
177  Base::m_t += normalized_waiting_time / cumulative_rate;
178 
179  //reduce remaining time of current time step
180  Base::m_dt -= normalized_waiting_time / cumulative_rate;
181 
182  //perform transition
183  Base::m_edge_func(event_edge.property, flat_index,
184  Base::m_graph.nodes()[event_edge.start_node_idx].property,
185  Base::m_graph.nodes()[event_edge.end_node_idx].property);
186 
187  //calculate new cumulative rate
188  cumulative_rate = get_cumulative_transition_rate();
189 
190  //draw new normalized waiting time
191  normalized_waiting_time = ExponentialDistribution<ScalarType>::get_instance()(1.0);
192 
193  } while (cumulative_rate * Base::m_dt > normalized_waiting_time);
194  }
195  else { //no transition event in current time step
196  normalized_waiting_time -= cumulative_rate * Base::m_dt; //reduce waiting time by current time step
197  }
198 
199  //advance nodes until t+dt
200  for (size_t node_iter = 0; node_iter < Base::m_graph.nodes().size(); ++node_iter) {
201  auto& node = Base::m_graph.nodes()[node_iter];
202  Base::m_node_func(Base::m_t, Base::m_dt, node.property);
203  //get new dt of each node
204  dt_cand[node_iter] = node.property.get_simulation().get_dt();
205  }
206 
207  //advance time
208  Base::m_t += Base::m_dt;
209  //new dt ist the minimal dt of all nodes
210  Base::m_dt = *std::min_element(dt_cand.begin(), dt_cand.end());
211  }
212  }
213 
214 private:
215  ScalarType get_cumulative_transition_rate()
216  {
217  //compute current cumulative transition rate
218  ScalarType cumulative_transition_rate = 0;
219  for (auto& e : Base::m_graph.edges()) {
220  cumulative_transition_rate +=
221  e.property.get_transition_rates(Base::m_graph.nodes()[e.start_node_idx].property).sum();
222  }
223  return cumulative_transition_rate;
224  }
225 
226  void get_rates(std::vector<ScalarType>& rates)
227  {
228  size_t j = 0;
229  for (auto& e : Base::m_graph.edges()) {
230  auto edge_rates = e.property.get_transition_rates(Base::m_graph.nodes()[e.start_node_idx].property);
231  for (Eigen::Index i = 0; i < edge_rates.size(); ++i) {
232  const auto compartment_value =
233  Base::m_graph.nodes()[e.start_node_idx].property.get_result().get_last_value()[i];
234  rates[j] = (compartment_value < 1.) ? 0. : edge_rates(i);
235 
236  j++;
237  }
238  }
239  }
